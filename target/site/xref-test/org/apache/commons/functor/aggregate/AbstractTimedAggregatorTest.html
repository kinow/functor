31  public class AbstractTimedAggregatorTest extends BaseFunctorTest {
32      /** Default timer interval used in the tests. */
33      private static final long DEFAULT_INTERVAL = 500L; // 0.5 second

35      /** Sleep extra msecs past the interval. */
36      private static final long SLEEP            = 10L;

43      @Test
44      public void testCreateNoTimer() throws Exception {
45          SimpleStoreTimedAggregator agg = (SimpleStoreTimedAggregator) makeFunctor();
46          assertEquals(agg.getInterval(), 0L);
47          assertFalse(agg.isTimerEnabled());
48          assertFalse(agg.isSharedTimer());
49          assertNull(agg.getTimerListeners());
50          assertFalse(agg.removeTimerListener(null)); // shouldn't NPE here
51                                                      // because the listeners
52                                                      // list is null!
53          agg.stop(); // shouldn't trigger any exceptions, even though there's no
54                      // timer
55      }

57      private void standardTimerTesting(SimpleStoreTimedAggregator agg, long interval) throws Exception {
58          final AtomicInteger count = new AtomicInteger();
59          TimedAggregatorListener<Integer> timedListener = new TimedAggregatorListener<Integer>() {
60              public void onTimer(AbstractTimedAggregator<Integer> aggregator, Integer evaluation) {
61                  count.incrementAndGet();
62              }
63          };
64  
65          assertEquals(agg.getInterval(), interval);
66          assertEquals(agg.getTimerListeners().size(), 0);
67          agg.addTimerListener(timedListener);
68          List<TimedAggregatorListener<Integer>> timerListeners = agg.getTimerListeners();
69          assertTrue(timerListeners.contains(timedListener));
70          TimeUnit.MILLISECONDS.sleep(interval + SLEEP);
71          // timer should have kicked in by now
72          assertTrue(count.intValue() > 0);
73  
74          TimedAggregatorListener<Integer> wrongListener = new TimedAggregatorListener<Integer>() {
75              public void onTimer(AbstractTimedAggregator<Integer> aggregator, Integer evaluation) {
76                  // do nothing in here
77              }
78          };
79          assertFalse(agg.removeTimerListener(wrongListener));
80          assertEquals(agg.getTimerListeners().size(), 1);
81          assertTrue(agg.removeTimerListener(timedListener));
82          int saveValue = count.get();
83          assertEquals(agg.getTimerListeners().size(), 0);
84          // give enough time for the timer to kick in again
85          TimeUnit.MILLISECONDS.sleep(interval + SLEEP);
86          assertEquals(saveValue, count.get());
87  
88      }

117     /**
118      * Simple timed aggregator which just stores the latest object and returns
119      * it when evaluating the result.
120      *
121      * @author Liviu Tudor http://about.me/liviutudor
122      */
123     class SimpleStoreTimedAggregator extends AbstractTimedAggregator<Integer> {
124         private Integer object;
125 
126         public SimpleStoreTimedAggregator() {
127             super();
128         }
129 
130         public SimpleStoreTimedAggregator(long interval) {
131             super(interval);
132         }
133 
134         public SimpleStoreTimedAggregator(long interval, boolean useSharedTimer) {
135             super(interval, useSharedTimer);
136         }
137 
138         @Override
139         protected void doAdd(Integer data) {
140             this.object = data;
141         }
142 
143         @Override
144         protected Integer doEvaluate() {
145             return object;
146         }
147 
148         @Override
149         protected int retrieveDataSize() {
150             return 1;
151         }
152 
153         @Override
154         protected void doReset() {
155             object = new Integer(0);
156         }
157     }
