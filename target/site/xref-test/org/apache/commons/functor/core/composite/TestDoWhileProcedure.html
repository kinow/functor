38  public class TestDoWhileProcedure extends BaseFunctorTest {
39  
40      // Functor Testing Framework
41      // ------------------------------------------------------------------------
42  
43      @Override
44      protected Object makeFunctor() {
45          return new DoWhileProcedure(NoOp.INSTANCE, Constant.FALSE);
46      }
47  
48      // Tests
49      // ------------------------------------------------------------------------
50      public class ListRemoveFirstProcedure implements Procedure {
51          protected List<Object> list;
52  
53  
54          public ListRemoveFirstProcedure(List<Object> list) {
55              this.list=list;
56          }
57  
58  
59          public void run() {
60              list.remove(0);
61          }
62      }
63  
64  
65      private List<Object> getList() {
66          List<Object> list = new LinkedList<Object>();
67          list.add("a");
68          list.add("b");
69          list.add("c");
70          list.add("d");
71          return list;
72      }
73  
74  
75      @Test
76      public void testLoopWithAction() throws Exception {
77          List<Object> list=getList();
78  
79          Procedure action=new ListRemoveFirstProcedure(list);
80          Predicate condition=new Not(new BoundPredicate(new IsEmpty<List<Object>>(), list));
81          Procedure procedure=new DoWhileProcedure(action, condition);
82  
83          assertTrue("The condition should be true before running the loop", condition.test());
84          assertFalse("The list should not be empty then", list.isEmpty());
85          procedure.run();
86          assertFalse("The condition should be false after running the loop", condition.test());
87          assertTrue("The list should be empty then", list.isEmpty());
88  
89          list=getList();
90          action=new ListRemoveFirstProcedure(list);
91          condition=new Predicate() {
92                        private int count=2;
93  
94                        public boolean test() {
95                            return count-- > 0;
96                        }
97                    };
98          procedure=new DoWhileProcedure(action, condition);
99          procedure.run();
100         assertFalse("The list should not contain \"a\" anymore", list.contains("a"));
101         assertFalse("The list should not contain \"b\" anymore", list.contains("b"));
102         assertFalse("The list should not contain \"c\" anymore", list.contains("c"));
103         assertTrue("The list should still contain \"d\"", list.contains("d"));
104     }
105 
106     @Test
107     public void testLoopForNothing() {
108         List<Object> list=getList();
109         Procedure action=new ListRemoveFirstProcedure(list);
110         Procedure procedure=new DoWhileProcedure(action, Constant.FALSE);
111         assertTrue("The list should contain 4 elements before runnning the loop", list.size()==4);
112         procedure.run();
113         assertTrue("The list should contain 3 elements after runnning the loop", list.size()==3);
114     }
