30  public class TestCompositeUnaryFunction extends BaseFunctorTest {

36      protected Object makeFunctor() {
37          return Composite.function(Constant.of(3));
38      }

40      // Tests
41      // ------------------------------------------------------------------------
42  
43      @Test
44      public void testEvaluate() throws Exception {
45  
46          assertEquals(new Integer(4),(new CompositeUnaryFunction<Object, Integer>(Constant.of(4))).evaluate(null));
47  
48          assertEquals(new Integer(4),(Composite.function(Constant.of(4)).of(Constant.of(3)).evaluate("xyzzy")));
49          assertEquals(new Integer(3),(new CompositeUnaryFunction<Object, Integer>(Constant.of(3)).of(Constant.of(4)).evaluate("xyzzy")));
50      }
51  
52      @Test
53      public void testOf() throws Exception {
54          UnaryFunction<Object, Integer> uf = new UnaryFunction<Object, Integer>() {
55              public Integer evaluate(Object obj) {
56                  if (obj instanceof Integer) {
57                      return (((Integer) obj).intValue()) + 1;
58                  } else {
59                      return 1;
60                  }
61              }
62          };
63          CompositeUnaryFunction<Object, Integer> f = null;
64          for (int i = 0; i < 10; i++) {
65              f = f == null ? new CompositeUnaryFunction<Object, Integer>(uf) : f.of(uf);
66              assertEquals(Integer.valueOf(i+1),f.evaluate(null));
67          }
68      }
69  
70      @Test
71      public void testEquals() throws Exception {
72          CompositeUnaryFunction<Object, String> f = new CompositeUnaryFunction<Object, String>(Constant.of("x"));
73          assertEquals(f,f);
74          
75          CompositeUnaryFunction<Object, String> g = new CompositeUnaryFunction<Object, String>(Constant.of("x"));
76          assertObjectsAreEqual(f,g);
77  
78          for (int i=0;i<3;i++) {
79              f = f.of(Constant.of("y")).of(Constant.of("z"));
80              assertObjectsAreNotEqual(f,g);
81              g = g.of(Constant.of("y")).of(Constant.of("z"));
82              assertObjectsAreEqual(f,g);
83          }
84  
85          assertObjectsAreNotEqual(f, Constant.of("y"));
86      }
