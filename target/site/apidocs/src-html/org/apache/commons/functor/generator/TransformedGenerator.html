017    package org.apache.commons.functor.generator;
018    
019    import org.apache.commons.functor.UnaryFunction;
020    import org.apache.commons.functor.UnaryProcedure;
021    import org.apache.commons.lang3.Validate;
022    
023    /**
024     * Generator that transforms the elements of another Generator.
025     *
026     * @param <I> the type of elements held in the wrapped generator.
027     * @param <E> the type of elements held in this generator.
028     * @version $Revision: 1344780 $ $Date: 2012-05-31 12:43:49 -0300 (Thu, 31 May 2012) $
029     */
030    public class TransformedGenerator<I, E> extends BaseGenerator<E> {
031    
032        /**
033         * The UnaryFunction to apply to each element.
034         */
035        private final UnaryFunction<? super I, ? extends E> func;
036    
037        /**
038         * Create a new TransformedGenerator.
039         * @param wrapped Generator to transform
040         * @param func UnaryFunction to apply to each element
041         */
042        public TransformedGenerator(Generator<? extends I> wrapped, UnaryFunction<? super I, ? extends E> func) {
043            super(Validate.notNull(wrapped, "Generator argument was null"));
044            this.func = Validate.notNull(func, "UnaryFunction argument was null");
045        }
046    
047        /**
048         * {@inheritDoc}
049         */
050        public void run(final UnaryProcedure<? super E> proc) {
051            getWrappedGenerator().run(new UnaryProcedure<I>() {
052                public void run(I obj) {
053                    proc.run(func.evaluate(obj));
054                }
055            });
056        }
057    
058        /**
059         * {@inheritDoc}
060         */
061        @SuppressWarnings("unchecked")
062        @Override
063        protected Generator<? extends I> getWrappedGenerator() {
064            return (Generator<? extends I>) super.getWrappedGenerator();
065        }
066    
067        /**
068         * {@inheritDoc}
069         */
070        @Override
071        public boolean equals(Object obj) {
072            if (obj == this) {
073                return true;
074            }
075            if (!(obj instanceof TransformedGenerator<?, ?>)) {
076                return false;
077            }
078            TransformedGenerator<?, ?> other = (TransformedGenerator<?, ?>) obj;
079            return other.getWrappedGenerator().equals(getWrappedGenerator()) && other.func == func;
080        }
081    
082        /**
083         * {@inheritDoc}
084         */
085        @Override
086        public int hashCode() {
087            int result = "TransformedGenerator".hashCode();
088            result <<= 2;
089            Generator<?> gen = getWrappedGenerator();
090            result ^= gen == null ? 0 : gen.hashCode();
091            result <<= 2;
092            result ^= func.hashCode();
093            return result;
094        }
095    }
