017    package org.apache.commons.functor.core.collection;
018    
019    import java.util.Iterator;
020    import java.util.NoSuchElementException;
021    
022    import org.apache.commons.functor.UnaryPredicate;
023    import org.apache.commons.lang3.Validate;
024    
025    /**
026     * Iterator that filters another Iterator by only passing through those elements
027     * that are matched by a specified UnaryPredicate.
028     *
029     * @param <T> the {@link Iterator} generic type
030     * @version $Revision: 1234990 $ $Date: 2012-01-23 19:18:10 -0200 (Mon, 23 Jan 2012) $
031     * @author Rodney Waldhoff
032     */
033    public final class FilteredIterator<T> implements Iterator<T> {
034        // attributes
035        // ------------------------------------------------------------------------
036    
037        /**
038         * The predicate used to test this Iterator elements.
039         */
040        private final UnaryPredicate<? super T> predicate;
041        /**
042         * The wrapped iterator.
043         */
044        private final Iterator<? extends T> iterator;
045    
046        /**
047         * Reference to next element has to be returned by this iterator.
048         */
049        private T next = null;
050        /**
051         * Flag to mark this iterator has more elements or not.
052         */
053        private boolean nextSet = false;
054        /**
055         * Flag to mark current iterator element can be removed.
056         */
057        private boolean canRemove = false;
058    
059        // constructor
060        // ------------------------------------------------------------------------
061        /**
062         * Create a new FilteredIterator.
063         * @param iterator to filter
064         * @param predicate to apply
065         */
066        public FilteredIterator(Iterator<? extends T> iterator, UnaryPredicate<? super T> predicate) {
067            this.iterator = Validate.notNull(iterator, "Iterator argument was null");
068            this.predicate = Validate.notNull(predicate, "filtering UnaryPredicate argument was null");
069        }
070    
071        // iterator methods
072        // ------------------------------------------------------------------------
073    
074        /**
075         * {@inheritDoc}
076         * @see java.util.Iterator#hasNext()
077         */
078        public boolean hasNext() {
079            return nextSet || setNext();
080        }
081    
082        /**
083         * {@inheritDoc}
084         * @see java.util.Iterator#next()
085         */
086        public T next() {
087            if (hasNext()) {
088                return returnNext();
089            }
090            throw new NoSuchElementException();
091        }
092    
093        /**
094         * {@inheritDoc}
095         * @see java.util.Iterator#remove()
096         */
097        public void remove() {
098            if (canRemove) {
099                canRemove = false;
100                iterator.remove();
101            } else {
102                throw new IllegalStateException();
103            }
104        }
105    
106        /**
107         * {@inheritDoc}
108         */
109        @Override
110        public boolean equals(Object obj) {
111            if (obj == this) {
112                return true;
113            }
114            if (!(obj instanceof FilteredIterator<?>)) {
115                return false;
116            }
117            FilteredIterator<?> that = (FilteredIterator<?>) obj;
118            return predicate.equals(that.predicate) && iterator.equals(that.iterator);
119        }
120    
121        /**
122         * {@inheritDoc}
123         */
124        @Override
125        public int hashCode() {
126            int hash = "FilteredIterator".hashCode();
127            hash <<= 2;
128            hash ^= predicate.hashCode();
129            hash <<= 2;
130            hash ^= iterator.hashCode();
131            return hash;
132        }
133    
134        /**
135         * {@inheritDoc}
136         */
137        @Override
138        public String toString() {
139            return "FilteredIterator<" + iterator + "," + predicate + ">";
140        }
141    
142        // static methods
143        // ------------------------------------------------------------------------
144        /**
145         * Get a filtered Iterator instance applying <code>pred</code> to <code>iter</code>.
146         * @param <T> the input iterator generic type
147         * @param iter to filter
148         * @param pred to apply
149         * @return Iterator
150         */
151        @SuppressWarnings("unchecked")
152        public static <T> Iterator<T> filter(Iterator<? extends T> iter, UnaryPredicate<? super T> pred) {
153            return null == pred ? (Iterator<T>) iter : (null == iter ? null : new FilteredIterator<T>(iter, pred));
154        }
155    
156        // private
157        // ------------------------------------------------------------------------
158        /**
159         * Set next element.
160         * @return whether the current iterator position is valid
161         */
162        private boolean setNext() {
163            while (iterator.hasNext()) {
164                canRemove = false;
165                T obj = iterator.next();
166                if (predicate.test(obj)) {
167                    nextSet = true;
168                    next = obj;
169                    return true;
170                }
171            }
172            next = null;
173            nextSet = false;
174            return false;
175        }
176    
177        /**
178         * Get the next element.
179         * @return next element.
180         */
181        private T returnNext() {
182            T temp = next;
183            canRemove = true;
184            next = null;
185            nextSet = false;
186            return temp;
187        }
188    
189    }
