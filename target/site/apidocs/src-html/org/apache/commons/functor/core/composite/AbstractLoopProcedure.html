017    package org.apache.commons.functor.core.composite;
018    
019    import org.apache.commons.functor.Predicate;
020    import org.apache.commons.functor.Procedure;
021    
022    import java.io.Serializable;
023    
024    /**
025     * Abstract base class for {@link WhileDoProcedure} and {@link DoWhileProcedure}
026     * used to implement loop procedures.
027     * <p>
028     * @version $Revision: 1187618 $ $Date: 2011-10-21 23:16:16 -0200 (Fri, 21 Oct 2011) $
029     * @author Herve Quiroz
030     * @author Rodney Waldhoff
031     */
032    public abstract class AbstractLoopProcedure implements Procedure, Serializable {
033        /**
034         * serialVersionUID declaration.
035         */
036        private static final long serialVersionUID = -5903381842630236070L;
037    
038        /** Base hash integer used to shift hash. */
039        private static final int HASH_SHIFT = 4;
040    
041        /**
042         * The condition has to be verified that while true,
043         * forces the action repetition.
044         */
045        private final Predicate condition;
046    
047        /**
048         * The action to be repeated until the condition is true.
049         */
050        private final Procedure action;
051    
052        /**
053         * Create a new AbstractLoopProcedure.
054         * @param condition while true, repeat
055         * @param action loop body
056         */
057        protected AbstractLoopProcedure(Predicate condition, Procedure action) {
058            this.condition = condition;
059            this.action = action;
060        }
061    
062        /**
063         * {@inheritDoc}
064         */
065        @Override
066        public final boolean equals(Object object) {
067            if (object == this) {
068                return true;
069            }
070            if (!(object instanceof AbstractLoopProcedure)) {
071                return false;
072            }
073            AbstractLoopProcedure that = (AbstractLoopProcedure) object;
074            return (null == getCondition() ? null == that.getCondition() : getCondition().equals(that.getCondition()))
075                    && (null == getAction() ? null == that.getAction() : getAction().equals(that.getAction()));
076        }
077    
078        /**
079         * {@inheritDoc}
080         */
081        @Override
082        public int hashCode() {
083            return hashCode("AbstractLoopProcedure".hashCode());
084        }
085    
086        /**
087         * {@inheritDoc}
088         */
089        @Override
090        public String toString() {
091            return getClass().getName() + "<" + getCondition() + "," + getAction() + ">";
092        }
093    
094        /**
095         * Create a hashCode by manipulating an input hashCode and factoring in instance state.
096         * @param hash incoming hashCode
097         * @return int
098         */
099        protected int hashCode(int hash) {
100            hash <<= HASH_SHIFT;
101            if (null != getAction()) {
102                hash ^= getAction().hashCode();
103            }
104            hash <<= HASH_SHIFT;
105            if (null != getCondition()) {
106                hash ^= getCondition().hashCode();
107            }
108            return hash;
109        }
110    
111        /**
112         * Get the condition.
113         * @return Predicate
114         */
115        protected final Predicate getCondition() {
116            return condition;
117        }
118    
119        /**
120         * Get the action.
121         * @return Procedure
122         */
123        protected final Procedure getAction() {
124            return action;
125        }
126    
127    }
