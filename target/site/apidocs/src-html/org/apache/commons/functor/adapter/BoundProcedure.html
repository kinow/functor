017    package org.apache.commons.functor.adapter;
018    
019    import java.io.Serializable;
020    
021    import org.apache.commons.functor.Procedure;
022    import org.apache.commons.functor.UnaryProcedure;
023    import org.apache.commons.lang3.Validate;
024    
025    /**
026     * Adapts a
027     * {@link UnaryProcedure UnaryProcedure}
028     * to the
029     * {@link Procedure Procedure} interface
030     * using a constant unary argument.
031     * <p/>
032     * Note that although this class implements
033     * {@link Serializable}, a given instance will
034     * only be truly <code>Serializable</code> if the
035     * underlying objects are.  Attempts to serialize
036     * an instance whose delegates are not
037     * <code>Serializable</code> will result in an exception.
038     *
039     * @version $Revision: 1234990 $ $Date: 2012-01-23 19:18:10 -0200 (Mon, 23 Jan 2012) $
040     * @author Rodney Waldhoff
041     */
042    public final class BoundProcedure implements Procedure, Serializable {
043        /**
044         * serialVersionUID declaration.
045         */
046        private static final long serialVersionUID = -6010802933400471747L;
047        /** The {@link UnaryProcedure UnaryProcedure} I'm wrapping. */
048        private final UnaryProcedure<Object> procedure;
049        /** The parameter to pass to {@code procedure}. */
050        private final Object param;
051    
052        /**
053         * Create a new BoundProcedure instance.
054         * @param <A> arg type
055         * @param procedure the procedure to adapt
056         * @param arg the constant argument to use
057         */
058        @SuppressWarnings("unchecked")
059        public <A> BoundProcedure(UnaryProcedure<? super A> procedure, A arg) {
060            this.procedure =
061                (UnaryProcedure<Object>) Validate.notNull(procedure,
062                    "UnaryProcedure argument was null");
063            this.param = arg;
064        }
065    
066        /**
067         * {@inheritDoc}
068         */
069        public void run() {
070            procedure.run(param);
071        }
072    
073        /**
074         * {@inheritDoc}
075         */
076        @Override
077        public boolean equals(Object that) {
078            return that == this || (that instanceof BoundProcedure && equals((BoundProcedure) that));
079        }
080    
081        /**
082         * Learn whether a given BoundProcedure is equal to this.
083         * @param that the BoundProcedure to test
084         * @return boolean
085         */
086        public boolean equals(BoundProcedure that) {
087            return null != that
088                    && (null == procedure ? null == that.procedure : procedure.equals(that.procedure))
089                    && (null == param ? null == that.param : param.equals(that.param));
090        }
091    
092        /**
093         * {@inheritDoc}
094         */
095        @Override
096        public int hashCode() {
097            int hash = "BoundProcedure".hashCode();
098            if (null != procedure) {
099                hash <<= 2;
100                hash ^= procedure.hashCode();
101            }
102            if (null != param) {
103                hash <<= 2;
104                hash ^= param.hashCode();
105            }
106            return hash;
107        }
108    
109        /**
110         * {@inheritDoc}
111         */
112        @Override
113        public String toString() {
114            return "BoundProcedure<" + procedure + "(" + param + ")>";
115        }
116    
117        /**
118         * Adapt the given, possibly-<code>null</code>,
119         * {@link UnaryProcedure UnaryProcedure} to the
120         * {@link Procedure Procedure} interface by binding
121         * the specified <code>Object</code> as a constant
122         * argument.
123         * When the given <code>UnaryProcedure</code> is <code>null</code>,
124         * returns <code>null</code>.
125         *
126         * @param <A> arg type
127         * @param procedure the possibly-<code>null</code>
128         *        {@link UnaryProcedure UnaryProcedure} to adapt
129         * @param arg the object to bind as a constant argument
130         * @return a <code>BoundProcedure</code> wrapping the given
131         *         {@link UnaryProcedure UnaryProcedure}, or <code>null</code>
132         *         if the given <code>UnaryProcedure</code> is <code>null</code>
133         */
134        public static <A> BoundProcedure bind(UnaryProcedure<? super A> procedure, A arg) {
135            return null == procedure ? null : new BoundProcedure(procedure, arg);
136        }
137    
138    }
