017    package org.apache.commons.functor;
018    
019    /**
020     * A functor that takes no arguments and returns no value.
021     * <p>
022     * Note that this functor implements the
023     * {@link Runnable Runnable}
024     * interface, making all <code>Procedure</code>s
025     * immediately usable in a number of contexts (Swing, Jelly, etc.).
026     * </p>
027     * <p>
028     * Implementors are encouraged but not required to make their functors
029     * {@link java.io.Serializable Serializable}.
030     * </p>
031     *
032     * @since 1.0
033     * @version $Revision: 647290 $ $Date: 2008-04-11 17:08:44 -0300 (Fri, 11 Apr 2008) $
034     * @author Rodney Waldhoff
035     */
036    public interface Procedure extends NullaryFunctor, Runnable {
037        /** Execute this procedure. */
038        void run();
039    }
