017    package org.apache.commons.functor.aggregate;
018    
019    import java.util.Collections;
020    import java.util.List;
021    import java.util.Timer;
022    import java.util.TimerTask;
023    import java.util.concurrent.CopyOnWriteArrayList;
024    import java.util.concurrent.locks.ReadWriteLock;
025    import java.util.concurrent.locks.ReentrantReadWriteLock;
026    
027    /**
028     * An aggregator which automatically resets the aggregated data at regular
029     * intervals and sends a notification when it is about to do so, so listeners
030     * can decide to gather the information before it is being reset (and log it
031     * etc). This allows for smaller memory footprints for instance in the case of
032     * List-based aggregators, as regularly the list is emptied. Also it allows for
033     * the call to <code>evaluate</code> to represent an "aggregated" value over a
034     * certain period of time. Note that you can still have a regular aggregator
035     * extending this class by specifying an interval less than or equal to zero.
036     * The regular flush/reset will be triggered from a timer which will always be
037     * started as a daemon thread (so it will stop when there are no more non-daemon
038     * threads in the JVM); this class allows 2 types of timers:
039     * <ul>
040     * <li>(default) per instance <code>Timer</code> -- each instance of this class
041     * will create a new <code>Timer</code> and this <code>Timer</code> will have a
042     * single <code>TimerTask</code> scheduled, which is the one that resets this
043     * <code>Aggregator</code> regularly and sends notifications. This way, when the
044     * <code>Aggregator</code> instance is destroyed, the <code>Timer</code> goes as
045     * well.</li>
046     * <li>shared <code>Timer</code> instance -- this class will create a static
047     * instance of <code>Timer</code> which can be shared by other instances of this
048     * class. While this is a bit more effective from a memory and thread management
049     * point of view, it has the downside that if the <code>TimerTask</code>'s are
050     * not managed properly this can create memory leaks. So if you decide to take
051     * this route make sure when you are finished with this instance, to always stop
052     * the timer at the end.</li>
053     * </ul>
054     * <br/>
055     * <p>
056     * <b>Synchronization</b>: This class provides a thread safe framework so when
057     * {@link #doAdd(Object)}, {@link #reset()} and {@link #evaluate()} is called,
058     * access is synchronized via a read-write lock. {@link #evaluate()} is
059     * considered a read operation and {@link #doAdd(Object)} and {@link #reset()}
060     * are considered write operations.
061     * </p>
062     *
063     * @param <T>
064     *            type of data to aggregate
065     */
066    public abstract class AbstractTimedAggregator<T> implements Aggregator<T> {
067        /**
068         * Stores a list to all objects which are listening for time events
069         * generated by this object. If there is no timer programmed (e.g.
070         * {@link #interval} is set to 0) this list will be <code>null</code>. Under
071         * the cover, this will use a <code>CopyOnWriteArrayList</code> since there
072         * aren't too many updates expected to this list.
073         *
074         * @see #interval
075         * @see #timer
076         * @see TimedAggregatorListener
077         */
078        private List<TimedAggregatorListener<T>> timerListeners;
079    
080        /**
081         * As per {@link #timer} javadoc, if the interval specified is zero or less
082         * there will be no <code>Timer</code> created/assigned to this instance.
083         * This constant is defined to make it easier to read code which creates
084         * instances of this class and doesn't assign them a timer.
085         */
086        public static final long                 NO_TIMER   = 0L;
087    
088        /**
089         * Name of the shared timer which will run all the TimerTasks resulted from
090         * creating instances of this class which are set to used the shared timer.
091         * This is useful when looking at thread dumps. For instances which use
092         * their own timer task, the name will be
093         * <code>TIMER_NAME + hashCode()</code>.
094         */
095        public static final String               TIMER_NAME = "TimedSummarizerMainTimer";
096    
097        /**
098         * The main shared timer which will execute all the <code>TimerTasks</code>
099         * resulted from instances of this class which chose to use the shared
100         * timer. Note that this <code>Timer</code> is started as a daemon thread so
101         * it will stop when there are no more non-daemon threads.
102         *
103         * @see #timer
104         */
105        private static final Timer               MAIN_TIMER = new Timer(TIMER_NAME, true);
106    
107        /**
108         * The timer instance for this instance. Can point to {@link #MAIN_TIMER} if
109         * shared timer was chosen in constructor or a newly created instance of
110         * <code>Timer</code> which is private to this instance only.
111         *
112         * @see #MAIN_TIMER
113         */
114        private Timer                            timer;
115    
116        /**
117         * Interval in milliseconds we flush the result of the "summary". This will
118         * be used to set up our <code>TimerTask</code> and schedule it with the
119         * <code>Timer</code>. Every time the timer kicks in after this interval, it
120         * will call {@link #timer()}. If this is set to a value of zero or less, no
121         * timer will be created.
122         */
123        private long                             interval;
124    
125        /**
126         * This is the task that is created when a new instance of this class is
127         * created. Once created this task will be scheduled with the {@link #timer}
128         * . Calling {@link #stop()} cancels this task and also will set it to null
129         * (so it can be recycled by the garbage collection), otherwise, until that
130         * point this will store a reference to a valid <code>TimerTask</code>
131         * instance.
132         */
133        private TimerTask                        task;
134    
135        /**
136         * Lock used internally to synchronize access to {@link #add(Object)},
137         * {@link #reset()} and {@link #evaluate()}. Locks for writing when
138         * {@link #add(Object)} and {@link #reset()} is called and for reading when
139         * {@link #evaluate()} is called.
140         *
141         * @see #add(Object)
142         * @see #evaluate()
143         * @see #reset()
144         */
145        private ReadWriteLock                    dataLock;
146    
147        /**
148         * Default constructor -- creates an instance of this aggregator with no
149         * <code>Timer</code>. Equivalent to
150         * <code>AbstractTimedAggregator(NO_TIMER)</code>.
151         *
152         * @see #AbstractTimedAggregator(long)
153         */
154        public AbstractTimedAggregator() {
155            this(NO_TIMER);
156        }
157    
158        /**
159         * Creates an aggregator which has a timer at the specified interval
160         * (miliseconds) and uses its own timer rather than the shared
161         * {@link #MAIN_TIMER}. Equivalent to
162         * <code>AbstractTimedAggregator(interval,false)</code>.
163         *
164         * @param interval
165         *            interval in miliseconds to set the timer for.
166         * @see #interval
167         * @see #timer
168         * @see #AbstractTimedAggregator(long, boolean)
169         */
170        public AbstractTimedAggregator(long interval) {
171            this(interval, false);
172        }
173    
174        /**
175         * Creates an aggregator which has a timer at the specified interval and
176         * also allows control over using the {@link #MAIN_TIMER shared timer} or
177         * its own per-instance timer.
178         *
179         * @param interval
180         *            interval in miliseconds to set the timer for.
181         * @param useSharedTimer
182         *            if set to <code>true</code>, {@link #timer} will be set to
183         *            {@link #TIMER_NAME}, otherwise a new instance of
184         *            <code>Timer</code> will be created.
185         */
186        public AbstractTimedAggregator(long interval, boolean useSharedTimer) {
187            if (interval <= NO_TIMER) {
188                // not using timer
189                this.interval = NO_TIMER;
190                this.timer = null;
191                this.task = null;
192                this.timerListeners = null;
193            } else {
194                // we have been requested to use timers
195                this.interval = interval;
196                this.timerListeners = new CopyOnWriteArrayList<TimedAggregatorListener<T>>();
197                if (useSharedTimer) {
198                    this.timer = MAIN_TIMER;
199                } else {
200                    this.timer = new Timer(TIMER_NAME + hashCode(), true);
201                }
202                // having set up the timer, create the task
203                this.task = new TimerTask() {
204                    @Override
205                    public void run() {
206                        timer();
207                    }
208                };
209                this.timer.scheduleAtFixedRate(this.task, this.interval, this.interval);
210            }
211            this.dataLock = new ReentrantReadWriteLock();
212        }
213    
214        /**
215         * Getter for {@link #interval}.
216         *
217         * @return Current value of {@link #interval}.
218         */
219        public final long getInterval() {
220            return interval;
221        }
222    
223        /**
224         * Adds the data to this aggregator. This function first locks
225         * {@link #dataLock} for writing then calls {@link #doAdd(Object)}, which
226         * allows subclasses to perform the actual adding to the aggregator and then
227         * at the end it unlocks {@link #dataLock}.
228         *
229         * @param data
230         *            Data to be added to the aggregator.
231         * @see #doAdd(Object)
232         * @see #dataLock
233         */
234        public final void add(T data) {
235            dataLock.writeLock().lock();
236            try {
237                doAdd(data);
238            } finally {
239                dataLock.writeLock().unlock();
240            }
241        }
242    
243        /**
244         * Function provided to allow subclasses to perform the actual adding of the
245         * data to the aggregator. This function is wrapped by {@link #add(Object)}
246         * so that access to any internal data series (implemented by subclasses)
247         * via {@link #add(Object)} or {@link #evaluate()} or {@link #reset()} is
248         * prohibited during this call, as a <b>write</b> lock is acquired prior to
249         * this function call to ensure this function is the only one which has
250         * access to the data.
251         *
252         * @param data
253         *            Data to be aggregated
254         * @see #add(Object)
255         */
256        protected abstract void doAdd(T data);
257    
258        /**
259         * Aggregates all the data this object has been "fed" via calls to
260         * {@link #add(Object)}. Note that this object delegates the call to
261         * {@link #doEvaluate()} after it secured read-only access to
262         * {@link #dataLock} -- so any data series access can be safely read
263         * (however, subclasses should NOT try to modify any data series they might
264         * implement at this point!). The lock is released after
265         * {@link #doEvaluate()} returns.
266         *
267         * @return result of aggregating the data, as returned by
268         *         {@link #doEvaluate()}
269         * @see #doEvaluate()
270         */
271        public final T evaluate() {
272            dataLock.readLock().lock();
273            try {
274                return doEvaluate();
275            } finally {
276                dataLock.readLock().unlock();
277            }
278        }
279    
280        /**
281         * Allows subclasses to perform the actual evaluation of the aggregated
282         * result in a thread-safe manner. When this function is called,
283         * <b>write</b> access to data (via {@link #add(Object)} and
284         * {@link #reset()}) is prohibited until this function finishes. However,
285         * please note that other read access (via calls to the same
286         * {@link #evaluate()}) is possible.
287         *
288         * @return Result of evaluating the aggregated data
289         */
290        protected abstract T doEvaluate();
291    
292        /**
293         * Resets this aggregator.This function first locks {@link #dataLock} for
294         * writing then calls {@link #doReset()}, which allows subclasses to perform
295         * the actual resetting of the aggregator and then at the end it unlocks
296         * {@link #dataLock}.
297         *
298         * @see #doReset()
299         */
300        public final void reset() {
301            dataLock.writeLock().lock();
302            try {
303                doReset();
304            } finally {
305                dataLock.writeLock().unlock();
306            }
307        }
308    
309        /**
310         * Function provided to allow subclasses to perform the actual reset of the
311         * aggregator. This function is wrapped by {@link #reset()} so that access
312         * to data (via {@link #add(Object)} or {@link #evaluate()} or
313         * {@link #reset()}) is prohibited during this call, as a <b>write</b> lock
314         * is acquired prior to this function call to ensure this function is the
315         * only one which has access to the data.
316         */
317        protected abstract void doReset();
318    
319        /**
320         * Retrieves the size of the currently-stored data series. This function
321         * first locks {@link #dataLock} for reading then calls
322         * {@link #retrieveDataSize()}, which allows subclasses to compute the data
323         * series size and then at the end it unlocks {@link #dataLock}.
324         *
325         * @return Size of the current data series, which will be aggregated at the
326         *         next call to {@link #evaluate()}
327         */
328        public final int getDataSize() {
329            dataLock.readLock().lock();
330            try {
331                return retrieveDataSize();
332            } finally {
333                dataLock.readLock().unlock();
334            }
335        }
336    
337        /**
338         * Function provided to allow subclasses to retrieve the actual size of the
339         * data series. This function is wrapped by {@link #getDataSize()} so that
340         * access to data (via {@link #add(Object)} or {@link #reset()}) is
341         * prohibited during this call, as a <b>read</b> lock is acquired prior to
342         * this function call. (However, calls to {@link #evaluate()} are allowed as
343         * that locks for reading too.)
344         *
345         * @return Size of the current data series. Zero means no data stored.
346         */
347        protected abstract int retrieveDataSize();
348    
349        /**
350         * Retrieves <b>an unmodifiable copy</b> of the {@link #timerListeners timer
351         * listeners}. Used for testing.
352         *
353         * @return <code>Collections.unmodifiableList(timerListeners)</code> if
354         *         {@link #timerListeners} is <b>not</b> <code>null</code>, or
355         *         <code>null</code> otherwise.
356         */
357        final List<TimedAggregatorListener<T>> getTimerListeners() {
358            if (timerListeners == null) {
359                return null;
360            }
361            return Collections.unmodifiableList(timerListeners);
362        }
363    
364        /**
365         * If this <code>Aggregator</code> has been started with timer support, it
366         * will add the given listener, so it receives
367         * {@link TimedAggregatorListener#onTimer(AbstractTimedAggregator,Object)
368         * timer events}. If no timer support has been configured for this
369         * Aggregator, this call has no effect.
370         *
371         * @param listener
372         *            Listener to be added to received timer events from this
373         *            aggregator.
374         * @see #timerListeners
375         */
376        public final void addTimerListener(TimedAggregatorListener<T> listener) {
377            if (timerListeners == null) {
378                return;
379            }
380            timerListeners.add(listener);
381        }
382    
383        /**
384         * Removes a listener from the timer listeners list if previously added. If
385         * this Aggregator has been configured with no timer support, this call will
386         * always return <code>false</code>.
387         *
388         * @param listener
389         *            Listener to be removed from the list. NullPointerException
390         *            thrown if this is null.
391         * @return <code>true</code> if this Aggregator has timer support and the
392         *         listener passed in was previously added (via
393         *         {@link #addTimerListener(TimedAggregatorListener)}) or false if
394         *         either the Aggregator has no timer support or it has timer
395         *         support but the listener was never registered with this
396         *         Aggregator.
397         * @see #timerListeners
398         */
399        public final boolean removeTimerListener(TimedAggregatorListener<T> listener) {
400            if (timerListeners == null) {
401                return false;
402            }
403            return timerListeners.remove(listener);
404        }
405    
406        /**
407         * Computes the current aggregated value (by calling {@link #evaluate()},
408         * resets this aggregator then notifies all listeners. Go through all the
409         * {@link #timerListeners} and sends
410         * {@link TimedAggregatorListener#onTimer(AbstractTimedAggregator,Object)
411         * notification messages} to each of them. Does nothing if
412         * {@link #timerListeners} is <code>null</code>. Please note that
413         * {@link #evaluate()} is called only once at the beginning of this
414         * function, and only if there are listeners configured, then this value is
415         * passed to every notification. This is in order to ensure all listeners
416         * receive the same value -- the value of the evaluation prior to resetting
417         * it.
418         */
419        private void timer() {
420            if (timerListeners != null) {
421                // if we have listeners, notify them
422                T aggregated = evaluate(); // NOTE: shouldn't evaluate() and reset()
423                                           // be done atomically here?
424                reset();
425                for (TimedAggregatorListener<T> i : timerListeners) {
426                    i.onTimer(this, aggregated);
427                }
428            } else {
429                reset();
430            }
431        }
432    
433        /**
434         * Checks whether this instance has a timer associated with it or not. If
435         * there is a timer for this Aggregator, then the {@link #task} member
436         * should be set to a non-null value.
437         *
438         * @return <code>true</code> if {@link #task} is not null,
439         *         <code>false</code> otherwise (in which case there is no timer).
440         */
441        public final boolean isTimerEnabled() {
442            return (task != null);
443        }
444    
445        /**
446         * Checks whether this instance uses its own timer or {@link #MAIN_TIMER the
447         * shared timer} for scheduling {@link #task the timer task}.
448         *
449         * @return <code>true</code> if <code>timer == MAIN_TIMER</code> or
450         *         <code>false</code> otherwise.
451         */
452        public final boolean isSharedTimer() {
453            return (timer == MAIN_TIMER);
454        }
455    
456        /**
457         * Cancels the current timer task (if set) -- which means from there on the
458         * data will not be reset anymore. Also, if {@link #timer} is not set to
459         * {@link #MAIN_TIMER the shared timer} then it will be cancelled as well
460         * Also releases all the listeners from the {@link #timerListeners list}.
461         */
462        public final void stop() {
463            // cancel the task first
464            if (task != null) {
465                task.cancel();
466                task = null;
467                timer.purge(); // remove the reference to this task
468            }
469            // then the timer if needed
470            if (timer != null && timer != MAIN_TIMER) {
471                timer.cancel();
472            }
473            timer = null;
474            // finally remove the elements from the listeners list
475            if (timerListeners != null) {
476                timerListeners.clear();
477            }
478        }
479    
480        @Override
481        protected final void finalize() throws Throwable {
482            // if we're going in the garbage, make sure we clean up
483            stop();
484        }
485    
486        @Override
487        public String toString() {
488            return AbstractTimedAggregator.class.getName();
489        }
