017    package org.apache.commons.functor.adapter;
018    
019    import java.io.Serializable;
020    
021    import org.apache.commons.functor.BinaryFunction;
022    import org.apache.commons.functor.BinaryProcedure;
023    import org.apache.commons.lang3.Validate;
024    
025    /**
026     * Adapts a
027     * {@link BinaryProcedure BinaryProcedure}
028     * to the
029     * {@link BinaryFunction BinaryFunction} interface
030     * by always returning <code>null</code>.
031     * <p/>
032     * Note that although this class implements
033     * {@link Serializable}, a given instance will
034     * only be truly <code>Serializable</code> if the
035     * underlying procedure is.  Attempts to serialize
036     * an instance whose delegate is not
037     * <code>Serializable</code> will result in an exception.
038     *
039     * @param <L> the left argument type.
040     * @param <R> the right argument type.
041     * @param <T> the returned value type.
042     * @version $Revision: 1234990 $ $Date: 2012-01-23 19:18:10 -0200 (Mon, 23 Jan 2012) $
043     * @author Rodney Waldhoff
044     */
045    public final class BinaryProcedureBinaryFunction<L, R, T> implements BinaryFunction<L, R, T>, Serializable {
046        /**
047         * serialVersionUID declaration.
048         */
049        private static final long serialVersionUID = 2099210273403668624L;
050        /** The {@link BinaryProcedure BinaryProcedure} I'm wrapping. */
051        private final BinaryProcedure<? super L, ? super R> procedure;
052    
053        /**
054         * Create a new BinaryProcedureBinaryFunction.
055         * @param procedure to adapt as a BinaryFunction
056         */
057        public BinaryProcedureBinaryFunction(BinaryProcedure<? super L, ? super R> procedure) {
058            this.procedure = Validate.notNull(procedure, "BinaryProcedure argument was null");
059        }
060    
061        /**
062         * {@inheritDoc}
063         */
064        public T evaluate(L left, R right) {
065            procedure.run(left, right);
066            return null;
067        }
068    
069        /**
070         * {@inheritDoc}
071         */
072        @Override
073        public boolean equals(Object that) {
074            return that == this || (that instanceof BinaryProcedureBinaryFunction<?, ?, ?>
075                                        && equals((BinaryProcedureBinaryFunction<?, ?, ?>) that));
076        }
077    
078        /**
079         * Learn whether another BinaryProcedureBinaryFunction is equal to this.
080         * @param that the BinaryProcedureBinaryFunction to test
081         * @return boolean
082         */
083        public boolean equals(BinaryProcedureBinaryFunction<?, ?, ?> that) {
084            return null != that && (null == procedure ? null == that.procedure : procedure.equals(that.procedure));
085        }
086    
087        /**
088         * {@inheritDoc}
089         */
090        @Override
091        public int hashCode() {
092            int hash = "BinaryProcedureBinaryFunction".hashCode();
093            if (null != procedure) {
094                hash ^= procedure.hashCode();
095            }
096            return hash;
097        }
098    
099        /**
100         * {@inheritDoc}
101         */
102        @Override
103        public String toString() {
104            return "BinaryProcedureBinaryFunction<" + procedure + ">";
105        }
106    
107        /**
108         * Adapt the given, possibly-<code>null</code>,
109         * {@link BinaryProcedure BinaryProcedure} to the
110         * {@link BinaryFunction BinaryFunction} interface.
111         * When the given <code>BinaryProcedure</code> is <code>null</code>,
112         * returns <code>null</code>.
113         *
114         * @param <L> the left argument type.
115         * @param <R> the right argument type.
116         * @param <T> the returned value type.
117         * @param procedure the possibly-<code>null</code>
118         *        {@link BinaryFunction BinaryFunction} to adapt
119         * @return a <code>BinaryProcedureBinaryFunction</code> wrapping the given
120         *         {@link BinaryFunction BinaryFunction}, or <code>null</code>
121         *         if the given <code>BinaryFunction</code> is <code>null</code>
122         */
123        public static <L, R, T> BinaryProcedureBinaryFunction<L, R, T> adapt(BinaryProcedure<? super L, ? super R> procedure) {
124            return null == procedure ? null : new BinaryProcedureBinaryFunction<L, R, T>(procedure);
125        }
126    
127    }
