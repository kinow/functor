017    package org.apache.commons.functor.core.composite;
018    
019    import java.io.Serializable;
020    import java.util.ArrayList;
021    import java.util.Iterator;
022    import java.util.List;
023    
024    import org.apache.commons.functor.Procedure;
025    
026    /**
027     * A {@link Procedure Procedure}
028     * that {@link Procedure#run runs} an ordered
029     * sequence of {@link Procedure Procedures}.
030     * When the sequence is empty, this procedure is does
031     * nothing.
032     * <p>
033     * Note that although this class implements
034     * {@link Serializable}, a given instance will
035     * only be truly <code>Serializable</code> if all the
036     * underlying functors are.  Attempts to serialize
037     * an instance whose delegates are not all
038     * <code>Serializable</code> will result in an exception.
039     * </p>
040     * @version $Revision: 1187618 $ $Date: 2011-10-21 23:16:16 -0200 (Fri, 21 Oct 2011) $
041     * @author Rodney Waldhoff
042     */
043    public class Sequence implements Procedure, Serializable {
044    
045        /**
046         * serialVersionUID declaration.
047         */
048        private static final long serialVersionUID = 8041703589149547883L;
049        // attributes
050        // ------------------------------------------------------------------------
051        /**
052         * The data structure where storing procedures sequence.
053         */
054        private List<Procedure> list = new ArrayList<Procedure>();
055    
056        // constructor
057        // ------------------------------------------------------------------------
058        /**
059         * Create a new Sequence.
060         */
061        public Sequence() {
062            super();
063        }
064    
065        /**
066         * Create a new Sequence instance.
067         *
068         * @param procedures to run sequentially
069         */
070        public Sequence(Procedure... procedures) {
071            this();
072            if (procedures != null) {
073                for (Procedure p : procedures) {
074                    then(p);
075                }
076            }
077        }
078    
079        /**
080         * Create a new Sequence instance.
081         *
082         * @param procedures to run sequentially
083         */
084        public Sequence(Iterable<Procedure> procedures) {
085            this();
086            if (procedures != null) {
087                for (Procedure p : procedures) {
088                    then(p);
089                }
090            }
091        }
092    
093        // modifiers
094        // ------------------------------------------------------------------------
095        /**
096         * Fluently add a Procedure.
097         * @param p Procedure to add
098         * @return this
099         */
100        public final Sequence then(Procedure p) {
101            list.add(p);
102            return this;
103        }
104    
105        // predicate interface
106        // ------------------------------------------------------------------------
107        /**
108         * {@inheritDoc}
109         */
110        public final void run() {
111            for (Iterator<Procedure> iter = list.iterator(); iter.hasNext();) {
112                iter.next().run();
113            }
114        }
115    
116        /**
117         * {@inheritDoc}
118         */
119        @Override
120        public final boolean equals(Object that) {
121            return that == this || (that instanceof Sequence && equals((Sequence) that));
122        }
123    
124        /**
125         * Learn whether a given Sequence is equal to this.
126         * @param that Sequence to test
127         * @return boolean
128         */
129        public boolean equals(Sequence that) {
130            // by construction, list is never null
131            return null != that && list.equals(that.list);
132        }
133    
134        /**
135         * {@inheritDoc}
136         */
137        @Override
138        public int hashCode() {
139            // by construction, list is never null
140            return "Sequence".hashCode() ^ list.hashCode();
141        }
142    
143        /**
144         * {@inheritDoc}
145         */
146        @Override
147        public String toString() {
148            return "Sequence<" + list + ">";
149        }
150    
151    }
