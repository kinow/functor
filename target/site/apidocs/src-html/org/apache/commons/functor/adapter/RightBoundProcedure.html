017    package org.apache.commons.functor.adapter;
018    
019    import java.io.Serializable;
020    
021    import org.apache.commons.functor.BinaryProcedure;
022    import org.apache.commons.functor.UnaryProcedure;
023    import org.apache.commons.lang3.Validate;
024    
025    /**
026     * Adapts a
027     * {@link BinaryProcedure BinaryProcedure}
028     * to the
029     * {@link UnaryProcedure UnaryProcedure} interface
030     * using a constant left-side argument.
031     * <p/>
032     * Note that although this class implements
033     * {@link Serializable}, a given instance will
034     * only be truly <code>Serializable</code> if the
035     * underlying objects are.  Attempts to serialize
036     * an instance whose delegates are not
037     * <code>Serializable</code> will result in an exception.
038     *
039     * @param <A> the argument type.
040     * @version $Revision: 1234990 $ $Date: 2012-01-23 19:18:10 -0200 (Mon, 23 Jan 2012) $
041     * @author Rodney Waldhoff
042     */
043    public final class RightBoundProcedure<A> implements UnaryProcedure<A>, Serializable {
044        /**
045         * serialVersionUID declaration.
046         */
047        private static final long serialVersionUID = 3267188080481758226L;
048        /** The {@link BinaryProcedure BinaryProcedure} I'm wrapping. */
049        private final BinaryProcedure<? super A, Object> procedure;
050        /** The parameter to pass to {@code procedure}. */
051        private final Object param;
052    
053        /**
054         * Create a new RightBoundProcedure.
055         * @param <R> bound arg type
056         * @param procedure the procedure to adapt
057         * @param arg the constant argument to use
058         */
059        @SuppressWarnings("unchecked")
060        public <R> RightBoundProcedure(BinaryProcedure<? super A, ? super R> procedure, R arg) {
061            this.procedure =
062                (BinaryProcedure<? super A, Object>) Validate.notNull(procedure,
063                    "BinaryProcedure argument was null");
064            this.param = arg;
065        }
066    
067        /**
068         * {@inheritDoc}
069         */
070        public void run(A obj) {
071            procedure.run(obj, param);
072        }
073    
074        /**
075         * {@inheritDoc}
076         */
077        @Override
078        public boolean equals(Object that) {
079            return that == this || (that instanceof RightBoundProcedure<?> && equals((RightBoundProcedure<?>) that));
080        }
081    
082        /**
083         * Learn whether another RightBoundProcedure is equal to this.
084         * @param that RightBoundProcedure to test
085         * @return boolean
086         */
087        public boolean equals(RightBoundProcedure<?> that) {
088            return null != that
089                    && (null == procedure ? null == that.procedure : procedure.equals(that.procedure))
090                    && (null == param ? null == that.param : param.equals(that.param));
091        }
092    
093        /**
094         * {@inheritDoc}
095         */
096        @Override
097        public int hashCode() {
098            int hash = "RightBoundProcedure".hashCode();
099            if (null != procedure) {
100                hash <<= 2;
101                hash ^= procedure.hashCode();
102            }
103            if (null != param) {
104                hash <<= 2;
105                hash ^= param.hashCode();
106            }
107            return hash;
108        }
109    
110        /**
111         * {@inheritDoc}
112         */
113        @Override
114        public String toString() {
115            return "RightBoundProcedure<" + procedure + "(?," + param + ")>";
116        }
117    
118        /**
119         * Get a UnaryProcedure from <code>procedure</code>.
120         * @param <L> the left argument type.
121         * @param <R> the right argument type.
122         * @param procedure to adapt
123         * @param arg right side argument
124         * @return RightBoundProcedure
125         */
126        public static <L, R> RightBoundProcedure<L> bind(BinaryProcedure<? super L, ? super R> procedure, R arg) {
127            return null == procedure ? null : new RightBoundProcedure<L>(procedure, arg);
128        }
129    
130    }
