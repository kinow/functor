017    package org.apache.commons.functor.adapter;
018    
019    import java.io.Serializable;
020    
021    import org.apache.commons.functor.BinaryPredicate;
022    import org.apache.commons.functor.UnaryPredicate;
023    import org.apache.commons.lang3.Validate;
024    
025    /**
026     * Adapts a
027     * {@link UnaryPredicate UnaryPredicate}
028     * to the
029     * {@link BinaryPredicate BinaryPredicate} interface
030     * by ignoring the second binary argument.
031     * <p/>
032     * Note that although this class implements
033     * {@link Serializable}, a given instance will
034     * only be truly <code>Serializable</code> if the
035     * underlying functor is.  Attempts to serialize
036     * an instance whose delegate is not
037     * <code>Serializable</code> will result in an exception.
038     *
039     * @param <L> the left argument type.
040     * @param <R> the right argument type.
041     * @version $Revision: 1234990 $ $Date: 2012-01-23 19:18:10 -0200 (Mon, 23 Jan 2012) $
042     * @author Rodney Waldhoff
043     */
044    public final class IgnoreRightPredicate<L, R> implements BinaryPredicate<L, R>, Serializable {
045        /**
046         * serialVersionUID declaration.
047         */
048        private static final long serialVersionUID = -4236624667788627722L;
049        /** The {@link UnaryPredicate UnaryPredicate} I'm wrapping. */
050        private final UnaryPredicate<? super L> predicate;
051    
052        /**
053         * Create a new IgnoreRightPredicate.
054         * @param predicate left
055         */
056        public IgnoreRightPredicate(UnaryPredicate<? super L> predicate) {
057            this.predicate = Validate.notNull(predicate, "UnaryPredicate argument was null");
058        }
059    
060        /**
061         * {@inheritDoc}
062         */
063        public boolean test(L left, R right) {
064            return predicate.test(left);
065        }
066    
067        /**
068         * {@inheritDoc}
069         */
070        @Override
071        public boolean equals(Object that) {
072            return that == this || (that instanceof IgnoreRightPredicate<?, ?>
073                                        && equals((IgnoreRightPredicate<?, ?>) that));
074        }
075    
076        /**
077         * Learn whether a given IgnoreRightPredicate is equal to this.
078         * @param that IgnoreRightPredicate to test
079         * @return boolean
080         */
081        public boolean equals(IgnoreRightPredicate<?, ?> that) {
082            return null != that && (null == predicate ? null == that.predicate : predicate.equals(that.predicate));
083        }
084    
085        /**
086         * {@inheritDoc}
087         */
088        @Override
089        public int hashCode() {
090            int hash = "IgnoreRightPredicate".hashCode();
091            if (null != predicate) {
092                hash ^= predicate.hashCode();
093            }
094            return hash;
095        }
096    
097        /**
098         * {@inheritDoc}
099         */
100        @Override
101        public String toString() {
102            return "IgnoreRightPredicate<" + predicate + ">";
103        }
104    
105        /**
106         * Adapt a UnaryPredicate as an IgnoreRightPredicate.
107         * @param <L> left type
108         * @param <R> right type
109         * @param predicate to adapt
110         * @return IgnoreRightPredicate<L, R>
111         */
112        public static <L, R> IgnoreRightPredicate<L, R> adapt(UnaryPredicate<? super L> predicate) {
113            return null == predicate ? null : new IgnoreRightPredicate<L, R>(predicate);
114        }
115    
116    }
