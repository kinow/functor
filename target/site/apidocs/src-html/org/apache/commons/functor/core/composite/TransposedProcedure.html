017    package org.apache.commons.functor.core.composite;
018    
019    import java.io.Serializable;
020    
021    import org.apache.commons.functor.BinaryProcedure;
022    import org.apache.commons.lang3.Validate;
023    
024    /**
025     * Transposes (swaps) the arguments to some other
026     * {@link BinaryProcedure procedure}.
027     * For example, given a procedure <i>p</i>
028     * and the ordered pair of arguments <i>a</i>,
029     * <i>b</i>.
030     * {@link #run runs}
031     * <code>p.run(b,a)</code>.
032     * <p>
033     * Note that although this class implements
034     * {@link Serializable}, a given instance will
035     * only be truly <code>Serializable</code> if the
036     * underlying functor is.  Attempts to serialize
037     * an instance whose delegate is not
038     * <code>Serializable</code> will result in an exception.
039     * </p>
040     * @param <L> the left argument type.
041     * @param <R> the right argument type.
042     * @version $Revision: 1234990 $ $Date: 2012-01-23 19:18:10 -0200 (Mon, 23 Jan 2012) $
043     * @author Rodney Waldhoff
044     */
045    public class TransposedProcedure<L, R> implements BinaryProcedure<L, R>, Serializable {
046        /**
047         * serialVersionUID declaration.
048         */
049        private static final long serialVersionUID = 4472683678460290015L;
050        // attributes
051        // ------------------------------------------------------------------------
052        /**
053         * The adapted procedure.
054         */
055        private final BinaryProcedure<? super R, ? super L> procedure;
056    
057        // constructor
058        // ------------------------------------------------------------------------
059        /**
060         * Create a new TransposedProcedure.
061         * @param procedure BinaryProcedure to transpose
062         */
063        public TransposedProcedure(BinaryProcedure<? super R, ? super L> procedure) {
064            this.procedure = Validate.notNull(procedure, "BinaryProcedure argument was null");
065        }
066    
067        // functor interface
068        // ------------------------------------------------------------------------
069        /**
070         * {@inheritDoc}
071         */
072        public void run(L left, R right) {
073            procedure.run(right, left);
074        }
075    
076        /**
077         * {@inheritDoc}
078         */
079        @Override
080        public boolean equals(Object that) {
081            return that == this || (that instanceof TransposedProcedure<?, ?> && equals((TransposedProcedure<?, ?>) that));
082        }
083    
084        /**
085         * Learn whether another TransposedProcedure is equal to this.
086         * @param that TransposedPredicate to test
087         * @return boolean
088         */
089        public boolean equals(TransposedProcedure<?, ?> that) {
090            return null != that && (null == procedure ? null == that.procedure : procedure.equals(that.procedure));
091        }
092    
093        /**
094         * {@inheritDoc}
095         */
096        @Override
097        public int hashCode() {
098            int hash = "TransposedProcedure".hashCode();
099            if (null != procedure) {
100                hash ^= procedure.hashCode();
101            }
102            return hash;
103        }
104    
105        /**
106         * {@inheritDoc}
107         */
108        @Override
109        public String toString() {
110            return "TransposedProcedure<" + procedure + ">";
111        }
112    
113        // static
114        // ------------------------------------------------------------------------
115        /**
116         * Transpose a BinaryProcedure.
117         *
118         * @param <L> the left argument type.
119         * @param <R> the right argument type.
120         * @param p to transpose
121         * @return TransposedProcedure
122         */
123        public static <L, R> TransposedProcedure<R, L> transpose(BinaryProcedure<? super L, ? super R> p) {
124            return null == p ? null : new TransposedProcedure<R, L>(p);
125        }
126    
127    }
