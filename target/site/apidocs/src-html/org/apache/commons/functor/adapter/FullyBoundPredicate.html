017    package org.apache.commons.functor.adapter;
018    
019    import java.io.Serializable;
020    
021    import org.apache.commons.functor.BinaryPredicate;
022    import org.apache.commons.functor.Predicate;
023    import org.apache.commons.lang3.Validate;
024    
025    /**
026     * Adapts a
027     * {@link BinaryPredicate BinaryPredicate}
028     * to the
029     * {@link org.apache.commons.functor.UnaryPredicate UnaryPredicate} interface
030     * using a constant left-side argument.
031     * <p/>
032     * Note that although this class implements
033     * {@link Serializable}, a given instance will
034     * only be truly <code>Serializable</code> if the
035     * underlying objects are.  Attempts to serialize
036     * an instance whose delegates are not
037     * <code>Serializable</code> will result in an exception.
038     *
039     * @version $Revision: 1234990 $ $Date: 2012-01-23 19:18:10 -0200 (Mon, 23 Jan 2012) $
040     * @author Matt Benson
041     */
042    public final class FullyBoundPredicate implements Predicate, Serializable {
043    
044        /**
045         * serialVersionUID declaration.
046         */
047        private static final long serialVersionUID = 7676235030688391839L;
048        /** The {@link BinaryPredicate BinaryPredicate} I'm wrapping. */
049        private final BinaryPredicate<Object, Object> predicate;
050        /** The left parameter to pass to {@code predicate}. */
051        private final Object left;
052        /** The right parameter to pass to {@code predicate}. */
053        private final Object right;
054    
055        /**
056         * Create a new FullyBoundPredicate instance.
057         * @param <L> left type
058         * @param <R> right type
059         * @param predicate the predicate to adapt
060         * @param left the left argument to use
061         * @param right the right argument to use
062         */
063        @SuppressWarnings("unchecked")
064        public <L, R> FullyBoundPredicate(BinaryPredicate<? super L, ? super R> predicate, L left, R right) {
065            this.predicate =
066                (BinaryPredicate<Object, Object>) Validate.notNull(predicate,
067                    "BinaryPredicate argument was null");
068            this.left = left;
069            this.right = right;
070        }
071    
072        /**
073         * {@inheritDoc}
074         */
075        public boolean test() {
076            return predicate.test(left, right);
077        }
078    
079        /**
080         * {@inheritDoc}
081         */
082        @Override
083        public boolean equals(Object that) {
084            return that == this || (that instanceof FullyBoundPredicate && equals((FullyBoundPredicate) that));
085        }
086    
087        /**
088         * Learn whether another FullyBoundPredicate is equal to this.
089         * @param that FullyBoundPredicate to test
090         * @return boolean
091         */
092        public boolean equals(FullyBoundPredicate that) {
093            return null != that && (null == predicate ? null == that.predicate : predicate.equals(that.predicate))
094                    && (null == left ? null == that.left : left.equals(that.left))
095                    && (null == right ? null == that.right : right.equals(that.right));
096        }
097    
098        /**
099         * {@inheritDoc}
100         */
101        @Override
102        public int hashCode() {
103            int hash = "FullyBoundPredicate".hashCode();
104            if (null != predicate) {
105                hash <<= 2;
106                hash ^= predicate.hashCode();
107            }
108            hash <<= 2;
109            if (null != left) {
110                hash ^= left.hashCode();
111            }
112            hash <<= 2;
113            if (null != right) {
114                hash ^= right.hashCode();
115            }
116            return hash;
117        }
118    
119        /**
120         * {@inheritDoc}
121         */
122        @Override
123        public String toString() {
124            return "FullyBoundPredicate<" + predicate + "(" + left + ", " + right + ")>";
125        }
126    
127        /**
128         * Adapt a BinaryPredicate to the Predicate interface.
129         * @param predicate to adapt
130         * @param <L> left type
131         * @param <R> right type
132         * @param left L argument to always send as the left operand to the wrapped function
133         * @param right R argument to always send as the right operand to the wrapped function
134         * @return FullyBoundPredicate
135         */
136        public static <L, R> FullyBoundPredicate bind(
137                BinaryPredicate<? super L, ? super R> predicate, L left, R right) {
138            return null == predicate ? null : new FullyBoundPredicate(predicate, left, right);
139        }
140    }
