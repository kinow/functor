017    package org.apache.commons.functor.core.composite;
018    
019    import java.io.Serializable;
020    import java.util.ArrayList;
021    import java.util.Iterator;
022    import java.util.List;
023    
024    import org.apache.commons.functor.UnaryProcedure;
025    
026    /**
027     * A {@link UnaryProcedure UnaryProcedure}
028     * that {@link UnaryProcedure#run runs} an ordered
029     * sequence of {@link UnaryProcedure UnaryProcedures}.
030     * When the sequence is empty, this procedure is does
031     * nothing.
032     * <p>
033     * Note that although this class implements
034     * {@link Serializable}, a given instance will
035     * only be truly <code>Serializable</code> if all the
036     * underlying functors are.  Attempts to serialize
037     * an instance whose delegates are not all
038     * <code>Serializable</code> will result in an exception.
039     * </p>
040     * @param <A> the argument type.
041     * @version $Revision: 1187618 $ $Date: 2011-10-21 23:16:16 -0200 (Fri, 21 Oct 2011) $
042     * @author Rodney Waldhoff
043     */
044    public class UnarySequence<A> implements UnaryProcedure<A>, Serializable {
045    
046        /**
047         * serialVersionUID declaration.
048         */
049        private static final long serialVersionUID = 9194268249717820246L;
050        // attributes
051        // ------------------------------------------------------------------------
052        /**
053         * The data structure to store the procedure sequence.
054         */
055        private final List<UnaryProcedure<? super A>> list = new ArrayList<UnaryProcedure<? super A>>();
056    
057        // constructor
058        // ------------------------------------------------------------------------
059        /**
060         * Create a new UnarySequence.
061         */
062        public UnarySequence() {
063            super();
064        }
065    
066        /**
067         * Create a new UnarySequence instance.
068         *
069         * @param procedures to run sequentially
070         */
071        public UnarySequence(UnaryProcedure<? super A>... procedures) {
072            this();
073            if (procedures != null) {
074                for (UnaryProcedure<? super A> p : procedures) {
075                    then(p);
076                }
077            }
078        }
079    
080        /**
081         * Create a new UnarySequence instance.
082         *
083         * @param procedures to run sequentially
084         */
085        public UnarySequence(Iterable<UnaryProcedure<? super A>> procedures) {
086            this();
087            if (procedures != null) {
088                for (UnaryProcedure<? super A> p : procedures) {
089                    then(p);
090                }
091            }
092        }
093    
094        // modifiers
095        // ------------------------------------------------------------------------
096        /**
097         * Fluently add a UnaryProcedure to the sequence.
098         * @param p UnaryProcedure to add
099         * @return this
100         */
101        public UnarySequence<A> then(UnaryProcedure<? super A> p) {
102            list.add(p);
103            return this;
104        }
105    
106        // predicate interface
107        // ------------------------------------------------------------------------
108        /**
109         * {@inheritDoc}
110         */
111        public void run(A obj) {
112            for (Iterator<UnaryProcedure<? super A>> iter = list.iterator(); iter.hasNext();) {
113                iter.next().run(obj);
114            }
115        }
116    
117        /**
118         * {@inheritDoc}
119         */
120        @Override
121        public boolean equals(Object that) {
122            return that == this || (that instanceof UnarySequence<?> && equals((UnarySequence<?>) that));
123        }
124    
125        /**
126         * Learn whether another UnarySequence is equal to this.
127         * @param that UnarySequence to test
128         * @return boolean
129         */
130        public boolean equals(UnarySequence<?> that) {
131            // by construction, list is never null
132            return null != that && list.equals(that.list);
133        }
134    
135        /**
136         * {@inheritDoc}
137         */
138        @Override
139        public int hashCode() {
140            // by construction, list is never null
141            return "UnarySequence".hashCode() ^ list.hashCode();
142        }
143    
144        /**
145         * {@inheritDoc}
146         */
147        @Override
148        public String toString() {
149            return "UnarySequence<" + list + ">";
150        }
151    
152    }
