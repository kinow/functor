017    package org.apache.commons.functor.core.composite;
018    
019    import java.io.Serializable;
020    
021    import org.apache.commons.functor.BinaryFunction;
022    import org.apache.commons.functor.BinaryPredicate;
023    import org.apache.commons.lang3.Validate;
024    
025    /**
026     * A {@link BinaryFunction BinaryFunction}
027     * similiar to Java's "ternary"
028     * or "conditional" operator (<code>&#x3F; &#x3A;</code>).
029     * Given a {@link BinaryPredicate predicate}
030     * <i>p</i> and {@link BinaryFunction functions}
031     * <i>f</i> and <i>g</i>, {@link #evaluate evaluates}
032     * to
033     * <code>p.test(x,y) ? f.evaluate(x,y) : g.evaluate(x,y)</code>.
034     * <p>
035     * Note that although this class implements
036     * {@link Serializable}, a given instance will
037     * only be truly <code>Serializable</code> if all the
038     * underlying functors are.  Attempts to serialize
039     * an instance whose delegates are not all
040     * <code>Serializable</code> will result in an exception.
041     * </p>
042     * @param <L> the left argument type.
043     * @param <R> the right argument type.
044     * @param <T> the output function returned value type.
045     * @version $Revision: 1234990 $ $Date: 2012-01-23 19:18:10 -0200 (Mon, 23 Jan 2012) $
046     * @author Rodney Waldhoff
047     */
048    public final class ConditionalBinaryFunction<L, R, T> implements BinaryFunction<L, R, T>, Serializable {
049        /**
050         * serialVersionUID declaration.
051         */
052        private static final long serialVersionUID = -994698971284481482L;
053    
054        /** Base hash integer used to shift hash. */
055        private static final int HASH_SHIFT = 4;
056        // attributes
057        // ------------------------------------------------------------------------
058        /**
059         * the condition to be evaluated.
060         */
061        private final BinaryPredicate<? super L, ? super R> ifPred;
062        /**
063         * the function executed if the condition is satisfied.
064         */
065        private final BinaryFunction<? super L, ? super R, ? extends T> thenFunc;
066        /**
067         * the function executed if the condition is not satisfied.
068         */
069        private final BinaryFunction<? super L, ? super R, ? extends T> elseFunc;
070    
071        // constructor
072        // ------------------------------------------------------------------------
073        /**
074         * Create a new ConditionalBinaryFunction.
075         * @param ifPred if
076         * @param thenFunc then
077         * @param elseFunc else
078         */
079        public ConditionalBinaryFunction(BinaryPredicate<? super L, ? super R> ifPred,
080                BinaryFunction<? super L, ? super R, ? extends T> thenFunc,
081                BinaryFunction<? super L, ? super R, ? extends T> elseFunc) {
082            this.ifPred = Validate.notNull(ifPred, "BinaryPredicate argument was null");
083            this.thenFunc = Validate.notNull(thenFunc, "'then' BinaryFunction argument was null");
084            this.elseFunc = Validate.notNull(elseFunc, "'else' BinaryFunction argument was null");
085        }
086    
087        // predicate interface
088        // ------------------------------------------------------------------------
089        /**
090         * {@inheritDoc}
091         */
092        public T evaluate(L left, R right) {
093            if (ifPred.test(left, right)) {
094                return thenFunc.evaluate(left, right);
095            } else {
096                return elseFunc.evaluate(left, right);
097            }
098        }
099    
100        /**
101         * {@inheritDoc}
102         */
103        @Override
104        public boolean equals(Object that) {
105            return that == this || (that instanceof ConditionalBinaryFunction<?, ?, ?>
106                                        && equals((ConditionalBinaryFunction<?, ?, ?>) that));
107        }
108    
109        /**
110         * Learn whether another ConditionalBinaryFunction is equal to this.
111         * @param that ConditionalBinaryFunction to test
112         * @return boolean
113         */
114        public boolean equals(ConditionalBinaryFunction<?, ?, ?> that) {
115            return null != that
116                    && (null == ifPred ? null == that.ifPred : ifPred.equals(that.ifPred))
117                    && (null == thenFunc ? null == that.thenFunc : thenFunc.equals(that.thenFunc))
118                    && (null == elseFunc ? null == that.elseFunc : elseFunc.equals(that.elseFunc));
119        }
120    
121        /**
122         * {@inheritDoc}
123         */
124        @Override
125        public int hashCode() {
126            int hash = "ConditionalBinaryFunction".hashCode();
127            if (null != ifPred) {
128                hash <<= HASH_SHIFT;
129                hash ^= ifPred.hashCode();
130            }
131            if (null != thenFunc) {
132                hash <<= HASH_SHIFT;
133                hash ^= thenFunc.hashCode();
134            }
135            if (null != elseFunc) {
136                hash <<= HASH_SHIFT;
137                hash ^= elseFunc.hashCode();
138            }
139            return hash;
140        }
141    
142        /**
143         * {@inheritDoc}
144         */
145        @Override
146        public String toString() {
147            return "ConditionalBinaryFunction<" + ifPred + "?" + thenFunc + ":" + elseFunc + ">";
148        }
149    
150    }
