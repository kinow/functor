017    package org.apache.commons.functor.adapter;
018    
019    import org.apache.commons.functor.BinaryPredicate;
020    import org.apache.commons.functor.UnaryPredicate;
021    import org.apache.commons.lang3.Validate;
022    
023    /**
024     * Adapts a BinaryPredicate as a UnaryPredicate by sending the same argument to both sides of the BinaryPredicate.
025     * @param <A> the argument type.
026     * @version $Revision: 1234990 $ $Date: 2012-01-23 19:18:10 -0200 (Mon, 23 Jan 2012) $
027     * @author Matt Benson
028     */
029    public final class BinaryPredicateUnaryPredicate<A> implements UnaryPredicate<A> {
030        /**
031         * The adapted {@link BinaryPredicate}.
032         */
033        private final BinaryPredicate<? super A, ? super A> predicate;
034    
035        /**
036         * Create a new BinaryPredicateUnaryPredicate.
037         * @param predicate to adapt
038         */
039        public BinaryPredicateUnaryPredicate(BinaryPredicate<? super A, ? super A> predicate) {
040            this.predicate = Validate.notNull(predicate, "BinaryPredicate argument was null");
041        }
042    
043        /**
044         * {@inheritDoc}
045         */
046        public boolean test(A obj) {
047            return predicate.test(obj, obj);
048        }
049    
050        /**
051         * {@inheritDoc}
052         */
053        @Override
054        public boolean equals(Object obj) {
055            return obj == this || obj instanceof BinaryPredicateUnaryPredicate<?>
056                    && equals((BinaryPredicateUnaryPredicate<?>) obj);
057        }
058    
059        /**
060         * Learn whether another BinaryPredicateUnaryPredicate is equal to
061         * <code>this</code>.
062         *
063         * @param that BinaryPredicateUnaryPredicate to check
064         *
065         * @return whether equal
066         */
067        public boolean equals(BinaryPredicateUnaryPredicate<?> that) {
068            return that != null && that.predicate.equals(this.predicate);
069        }
070    
071        /**
072         * {@inheritDoc}
073         */
074        @Override
075        public int hashCode() {
076            return ("BinaryPredicateUnaryPredicate".hashCode() << 2) | predicate.hashCode();
077        }
078    
079        /**
080         * {@inheritDoc}
081         */
082        @Override
083        public String toString() {
084            return "BinaryPredicateUnaryPredicate<" + predicate + ">";
085        }
086    
087        /**
088         * Adapt a BinaryFunction as a UnaryFunction.
089         * @param <A> the argument type.
090         * @param predicate BinaryPredicate to adapt
091         * @return UnaryPredicate<A>
092         */
093        public static <A> UnaryPredicate<A> adapt(BinaryPredicate<? super A, ? super A> predicate) {
094            return null == predicate ? null : new BinaryPredicateUnaryPredicate<A>(predicate);
095        }
096    
097    }
