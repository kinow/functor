015    package org.apache.commons.functor.generator.util;
016    
017    import java.util.Arrays;
018    import java.util.Iterator;
019    import java.util.Map;
020    
021    import org.apache.commons.functor.generator.Generator;
022    import org.apache.commons.functor.generator.IteratorToGeneratorAdapter;
023    
024    /**
025     * Generator factory for each element of a "collection".
026     *
027     * @since 1.0
028     * @version $Revision: 1234401 $ $Date: 2012-01-21 17:14:43 -0200 (Sat, 21 Jan 2012) $
029     * @author  Jason Horman (jason@jhorman.org)
030     */
031    public final class EachElement {
032        /**
033         * <p>{@code EachElement} instances should NOT be constructed in
034         * standard programming. Instead, the methods of the class should be invoked
035         * statically.</p>
036         *
037         * <p>This constructor is public to permit tools that require a JavaBean
038         * instance to operate.</p>
039         */
040        public EachElement() {
041        }
042    
043        /**
044         * Get a Generator for each element of an Iterable.
045         * @param <E> the type of elements held in the input iterable.
046         * @param iterable to iterate
047         * @return Generator<E>
048         */
049        public static <E> Generator<E> from(Iterable<? extends E> iterable) {
050            return iterable == null ? null : EachElement.from(iterable.iterator());
051        }
052    
053        /**
054         * Get a Generator for each entry of a Map.
055         * @param <K> the type of keys maintained by the input map.
056         * @param <V> the type of mapped values in the input map.
057         * @param map to iterate
058         * @return Generator
059         */
060        @SuppressWarnings("unchecked")
061        public static <K, V> Generator<Map.Entry<K, V>> from(Map<? extends K, ? extends V> map) {
062            return map == null ? null : EachElement.from(((Map<K, V>) map).entrySet().iterator());
063        }
064    
065        /**
066         * Get a Generator for each element of an Object[].
067         * @param <E> the type of elements held in the input array.
068         * @param array to iterate
069         * @return Generator
070         */
071        public static <E> Generator<E> from(E... array) {
072            return array == null ? null : EachElement.from(Arrays.asList(array).iterator());
073        }
074    
075        /**
076         * Get a Generator for each element of an Iterator.
077         * @param <E> the type of elements held in the input iterator.
078         * @param iter to iterate
079         * @return Generator
080         */
081        public static <E> Generator<E> from(Iterator<? extends E> iter) {
082            return iter == null ? null : new IteratorToGeneratorAdapter<E>(iter);
083        }
084    }
