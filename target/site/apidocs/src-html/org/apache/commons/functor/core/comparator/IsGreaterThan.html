017    package org.apache.commons.functor.core.comparator;
018    
019    import java.io.Serializable;
020    import java.util.Comparator;
021    
022    import org.apache.commons.functor.BinaryPredicate;
023    import org.apache.commons.functor.UnaryPredicate;
024    import org.apache.commons.functor.adapter.RightBoundPredicate;
025    import org.apache.commons.lang3.Validate;
026    
027    /**
028     * A {@link BinaryPredicate BinaryPredicate} that {@link #test tests}
029     * <code>true</code> iff the left argument is greater than the
030     * right argument under the specified {@link Comparator}.
031     * When no (or a <code>null</code> <code>Comparator</code> is specified,
032     * a {@link Comparable Comparable} <code>Comparator</code> is used.
033     *
034     * @param <T> the binary predicate input types
035     * @version $Revision: 1234990 $ $Date: 2012-01-23 19:18:10 -0200 (Mon, 23 Jan 2012) $
036     * @author Rodney Waldhoff
037     */
038    public final class IsGreaterThan<T> implements BinaryPredicate<T, T>, Serializable {
039    
040        /**
041         * Basic IsGreaterThan instance.
042         */
043        public static final IsGreaterThan<Comparable<?>> INSTANCE = IsGreaterThan.<Comparable<?>>instance();
044    
045        /**
046         * serialVersionUID declaration.
047         */
048        private static final long serialVersionUID = -8592932496891910273L;
049    
050        /**
051         * The wrapped comparator.
052         */
053        private final Comparator<? super T> comparator;
054    
055        /**
056         * Construct a <code>IsGreaterThan</code> {@link BinaryPredicate predicate}
057         * for {@link Comparable Comparable}s.
058         */
059        @SuppressWarnings("unchecked")
060        public IsGreaterThan() {
061            this(ComparableComparator.INSTANCE);
062        }
063    
064        /**
065         * Construct a <code>IsGreaterThan</code> {@link BinaryPredicate predicate}
066         * for the given {@link Comparator Comparator}.
067         *
068         * @param comparator the {@link Comparator Comparator}, when <code>null</code>,
069         *        a <code>Comparator</code> for {@link Comparable Comparable}s will
070         *        be used.
071         */
072        public IsGreaterThan(Comparator<? super T> comparator) {
073            this.comparator = Validate.notNull(comparator, "Comparator argument must not be null");
074        }
075    
076        /**
077         * Return <code>true</code> iff the <i>left</i> parameter is
078         * greater than the <i>right</i> parameter under my current
079         * {@link Comparator Comparator}.
080         * {@inheritDoc}
081         */
082        public boolean test(T left, T right) {
083            return comparator.compare(left, right) > 0;
084        }
085    
086        /**
087         * {@inheritDoc}
088         */
089        @Override
090        public boolean equals(Object that) {
091            return that == this || (that instanceof IsGreaterThan<?> && equals((IsGreaterThan<?>) that));
092        }
093    
094        /**
095         * Learn whether a given IsGreaterThan is equal to this.
096         * @param that the IsGreaterThan to test
097         * @return boolean
098         */
099        public boolean equals(IsGreaterThan<?> that) {
100            if (null != that) {
101                if (null == comparator) {
102                    return null == that.comparator;
103                }
104                return comparator.equals(that.comparator);
105            }
106            return false;
107        }
108    
109        /**
110         * {@inheritDoc}
111         */
112        @Override
113        public int hashCode() {
114            int hash = "IsGreaterThan".hashCode();
115            // by construction, comparator is never null
116            hash ^= comparator.hashCode();
117            return hash;
118        }
119    
120        /**
121         * {@inheritDoc}
122         */
123        @Override
124        public String toString() {
125            return "IsGreaterThan<" + comparator + ">";
126        }
127    
128        /**
129         * Get a typed IsGreaterThan instance.
130         *
131         * @param <T> the binary predicate input types
132         * @return IsGreaterThan<T>
133         */
134        public static <T extends Comparable<?>> IsGreaterThan<T> instance() {
135            return new IsGreaterThan<T>();
136        }
137    
138        /**
139         * Get an IsGreaterThan UnaryPredicate.
140         *
141         * @param <T> the binary predicate input types
142         * @param right the right side object of the IsGreaterThan comparison
143         * @return UnaryPredicate<T>
144         */
145        public static <T extends Comparable<?>> UnaryPredicate<T> instance(T right) {
146            return RightBoundPredicate.bind(new IsGreaterThan<T>(), right);
147        }
148    
149    }
