015    package org.apache.commons.functor.generator;
016    
017    import java.util.Collection;
018    
019    import org.apache.commons.functor.UnaryFunction;
020    import org.apache.commons.functor.generator.util.CollectionTransformer;
021    
022    /**
023     * Base class for generators. Adds support for all of the Algorithms to
024     * each subclass.
025     *
026     * @param <E> the type of elements held in this generator.
027     * @since 1.0
028     * @version $Revision: 1188397 $ $Date: 2011-10-24 19:40:54 -0200 (Mon, 24 Oct 2011) $
029     * @author  Jason Horman (jason@jhorman.org)
030     */
031    public abstract class BaseGenerator<E> implements Generator<E> {
032    
033        /** A generator can wrap another generator. */
034        private final Generator<?> wrappedGenerator;
035    
036        /** Set to true when the generator is {@link #stop stopped}. */
037        private boolean stopped = false;
038    
039        /** Create a new generator. */
040        public BaseGenerator() {
041            this(null);
042        }
043    
044        /**
045         * A generator can wrap another generator. When wrapping generators you
046         * should use probably this constructor since doing so will cause the
047         * {@link #stop} method to stop the wrapped generator as well.
048         * @param generator Generator to wrap
049         */
050        public BaseGenerator(Generator<?> generator) {
051            this.wrappedGenerator = generator;
052        }
053    
054        /**
055         * Get the generator that is being wrapped.
056         * @return Generator
057         */
058        protected Generator<?> getWrappedGenerator() {
059            return wrappedGenerator;
060        }
061    
062        /**
063         * {@inheritDoc}
064         * Stop the generator. Will stop the wrapped generator if one was set.
065         */
066        public void stop() {
067            if (wrappedGenerator != null) {
068                wrappedGenerator.stop();
069            }
070            stopped = true;
071        }
072    
073        /**
074         * {@inheritDoc}
075         * Check if the generator is stopped.
076         */
077        public boolean isStopped() {
078            return stopped;
079        }
080    
081        /**
082         * {@inheritDoc}
083         * Transforms this generator using the passed in
084         * UnaryFunction. An example function might turn the contents of the
085         * generator into a {@link Collection} of elements.
086         */
087        public final <T> T to(UnaryFunction<Generator<? extends E>, ? extends T> transformer) {
088            return transformer.evaluate(this);
089        }
090    
091        /**
092         * {@inheritDoc}
093         * Same as to(new CollectionTransformer(collection)).
094         */
095        public final Collection<? super E> to(Collection<? super E> collection) {
096            return to(new CollectionTransformer<E>(collection));
097        }
098    
099        /**
100         * {@inheritDoc}
101         * Same as to(new CollectionTransformer()).
102         */
103        @SuppressWarnings("unchecked")
104        public final Collection<E> toCollection() {
105            return (Collection<E>) to(new CollectionTransformer<E>());
106        }
107    }
