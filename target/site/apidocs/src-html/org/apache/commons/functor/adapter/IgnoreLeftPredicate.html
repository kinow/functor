017    package org.apache.commons.functor.adapter;
018    
019    import java.io.Serializable;
020    
021    import org.apache.commons.functor.BinaryPredicate;
022    import org.apache.commons.functor.UnaryPredicate;
023    import org.apache.commons.lang3.Validate;
024    
025    /**
026     * Adapts a
027     * {@link UnaryPredicate UnaryPredicate}
028     * to the
029     * {@link BinaryPredicate BinaryPredicate} interface
030     * by ignoring the first binary argument.
031     * <p/>
032     * Note that although this class implements
033     * {@link Serializable}, a given instance will
034     * only be truly <code>Serializable</code> if the
035     * underlying functor is.  Attempts to serialize
036     * an instance whose delegate is not
037     * <code>Serializable</code> will result in an exception.
038     *
039     * @param <L> the left argument type.
040     * @param <R> the right argument type.
041     * @version $Revision: 1234990 $ $Date: 2012-01-23 19:18:10 -0200 (Mon, 23 Jan 2012) $
042     * @author Rodney Waldhoff
043     */
044    public final class IgnoreLeftPredicate<L, R> implements BinaryPredicate<L, R>, Serializable {
045        /**
046         * serialVersionUID declaration.
047         */
048        private static final long serialVersionUID = -3200352647509255939L;
049        /** The {@link UnaryPredicate UnaryPredicate} I'm wrapping. */
050        private final UnaryPredicate<? super R> predicate;
051    
052        /**
053         * Create a new IgnoreLeftPredicate.
054         * @param predicate the right predicate
055         */
056        public IgnoreLeftPredicate(UnaryPredicate<? super R> predicate) {
057            this.predicate = Validate.notNull(predicate, "UnaryPredicate argument was null");
058        }
059    
060        /**
061         * {@inheritDoc}
062         */
063        public boolean test(L left, R right) {
064            return predicate.test(right);
065        }
066    
067        /**
068         * {@inheritDoc}
069         */
070        @Override
071        public boolean equals(Object that) {
072            return that == this || (that instanceof IgnoreLeftPredicate<?, ?> && equals((IgnoreLeftPredicate<?, ?>) that));
073        }
074    
075        /**
076         * Learn whether another IgnoreLeftPredicate is equal to this.
077         * @param that the IgnoreLeftPredicate to test
078         * @return boolean
079         */
080        public boolean equals(IgnoreLeftPredicate<?, ?> that) {
081            return null != that && (null == predicate ? null == that.predicate : predicate.equals(that.predicate));
082        }
083    
084        /**
085         * {@inheritDoc}
086         */
087        @Override
088        public int hashCode() {
089            int hash = "IgnoreLeftPredicate".hashCode();
090            if (null != predicate) {
091                hash ^= predicate.hashCode();
092            }
093            return hash;
094        }
095    
096        /**
097         * {@inheritDoc}
098         */
099        @Override
100        public String toString() {
101            return "IgnoreLeftPredicate<" + predicate + ">";
102        }
103    
104        /**
105         * Adapt a UnaryPredicate to an IgnoreLeftPredicate.
106         * @param <L> left type
107         * @param <R> right type
108         * @param predicate to adapt
109         * @return IgnoreLeftPredicate<L, R>
110         */
111        public static <L, R> IgnoreLeftPredicate<L, R> adapt(UnaryPredicate<? super R> predicate) {
112            return null == predicate ? null : new IgnoreLeftPredicate<L, R>(predicate);
113        }
114    
115    }
