017    package org.apache.commons.functor.core.composite;
018    
019    import java.io.Serializable;
020    
021    import org.apache.commons.functor.BinaryPredicate;
022    import org.apache.commons.functor.BinaryProcedure;
023    import org.apache.commons.functor.core.NoOp;
024    import org.apache.commons.lang3.Validate;
025    
026    /**
027     * A {@link BinaryProcedure BinaryProcedure}
028     * similiar to Java's "ternary"
029     * or "conditional" operator (<code>&#x3F; &#x3A;</code>).
030     * Given a {@link BinaryPredicate predicate}
031     * <i>p</i> and {@link BinaryProcedure procedures}
032     * <i>q</i> and <i>r</i>, {@link #run runs}
033     * <code>if (p.test(x,y)) { q.run(x,y); } else { r.run(x,y); }</code>.
034     * <p>
035     * Note that although this class implements
036     * {@link Serializable}, a given instance will
037     * only be truly <code>Serializable</code> if all the
038     * underlying functors are.  Attempts to serialize
039     * an instance whose delegates are not all
040     * <code>Serializable</code> will result in an exception.
041     * </p>
042     * @param <L> the left argument type.
043     * @param <R> the right argument type.
044     * @version $Revision: 1234990 $ $Date: 2012-01-23 19:18:10 -0200 (Mon, 23 Jan 2012) $
045     * @author Rodney Waldhoff
046     */
047    public final class ConditionalBinaryProcedure<L, R> implements BinaryProcedure<L, R>, Serializable {
048    
049        /**
050         * serialVersionUID declaration.
051         */
052        private static final long serialVersionUID = -3521992036791188475L;
053    
054        /** Base hash integer used to shift hash. */
055        private static final int HASH_SHIFT = 4;
056        // attributes
057        // ------------------------------------------------------------------------
058        /**
059         * the condition to be evaluated.
060         */
061        private final BinaryPredicate<? super L, ? super R> ifPred;
062        /**
063         * the predicate executed if the condition is satisfied.
064         */
065        private final BinaryProcedure<? super L, ? super R> thenProc;
066        /**
067         * the predicate executed if the condition is not satisfied.
068         */
069        private final BinaryProcedure<? super L, ? super R> elseProc;
070    
071        // constructor
072        // ------------------------------------------------------------------------
073    
074        /**
075         * Create a new ConditionalBinaryProcedure.
076         * @param ifPred to evaluate
077         * @param thenProc if <code>ifPred</code> yields <code>true</code>
078         */
079        public ConditionalBinaryProcedure(BinaryPredicate<? super L, ? super R> ifPred,
080                BinaryProcedure<? super L, ? super R> thenProc) {
081            this(ifPred, thenProc, NoOp.instance());
082        }
083    
084        /**
085         * Create a new ConditionalBinaryProcedure.
086         * @param ifPred to evaluate
087         * @param thenProc if <code>ifPred</code> yields <code>true</code>
088         * @param elseProc if <code>ifPred</code> yields <code>false</code>
089         */
090        public ConditionalBinaryProcedure(BinaryPredicate<? super L, ? super R> ifPred,
091                BinaryProcedure<? super L, ? super R> thenProc, BinaryProcedure<? super L, ? super R> elseProc) {
092            this.ifPred = Validate.notNull(ifPred, "BinaryPredicate argument was null");
093            this.thenProc = Validate.notNull(thenProc, "'then' BinaryProcedure argument was null");
094            this.elseProc = Validate.notNull(elseProc, "'else' BinaryProcedure argument was null");
095        }
096    
097        // predicate interface
098        // ------------------------------------------------------------------------
099    
100        /**
101         * {@inheritDoc}
102         */
103        public void run(L left, R right) {
104            if (ifPred.test(left, right)) {
105                thenProc.run(left, right);
106            } else {
107                elseProc.run(left, right);
108            }
109        }
110    
111        /**
112         * {@inheritDoc}
113         */
114        @Override
115        public boolean equals(Object that) {
116            return that == this || (that instanceof ConditionalBinaryProcedure<?, ?>
117                                        && equals((ConditionalBinaryProcedure<?, ?>) that));
118        }
119    
120        /**
121         * Learn whether a given ConditionalBinaryProcedure is equal to this.
122         * @param that compared object
123         * @return boolean
124         */
125        public boolean equals(ConditionalBinaryProcedure<?, ?> that) {
126            return null != that
127                    && (null == ifPred ? null == that.ifPred : ifPred.equals(that.ifPred))
128                    && (null == thenProc ? null == that.thenProc : thenProc.equals(that.thenProc))
129                    && (null == elseProc ? null == that.elseProc : elseProc.equals(that.elseProc));
130        }
131    
132        /**
133         * {@inheritDoc}
134         */
135        @Override
136        public int hashCode() {
137            int hash = "ConditionalBinaryProcedure".hashCode();
138            if (null != ifPred) {
139                hash <<= HASH_SHIFT;
140                hash ^= ifPred.hashCode();
141            }
142            if (null != thenProc) {
143                hash <<= HASH_SHIFT;
144                hash ^= thenProc.hashCode();
145            }
146            if (null != elseProc) {
147                hash <<= HASH_SHIFT;
148                hash ^= elseProc.hashCode();
149            }
150            return hash;
151        }
152    
153        /**
154         * {@inheritDoc}
155         */
156        @Override
157        public String toString() {
158            return "ConditionalBinaryProcedure<" + ifPred + "?" + thenProc + ":" + elseProc + ">";
159        }
160    }
