017    package org.apache.commons.functor.adapter;
018    
019    import java.io.Serializable;
020    
021    import org.apache.commons.functor.Predicate;
022    import org.apache.commons.functor.UnaryPredicate;
023    import org.apache.commons.lang3.Validate;
024    
025    /**
026     * Adapts a
027     * {@link UnaryPredicate UnaryPredicate}
028     * to the
029     * {@link Predicate Predicate} interface
030     * using a constant unary argument.
031     * <p/>
032     * Note that although this class implements
033     * {@link Serializable}, a given instance will
034     * only be truly <code>Serializable</code> if the
035     * underlying objects are.  Attempts to serialize
036     * an instance whose delegates are not
037     * <code>Serializable</code> will result in an exception.
038     *
039     * @version $Revision: 1234990 $ $Date: 2012-01-23 19:18:10 -0200 (Mon, 23 Jan 2012) $
040     * @author Rodney Waldhoff
041     */
042    public final class BoundPredicate implements Predicate, Serializable {
043        /**
044         * serialVersionUID declaration.
045         */
046        private static final long serialVersionUID = -5721164265625291834L;
047        /** The {@link UnaryPredicate UnaryPredicate} I'm wrapping. */
048        private final UnaryPredicate<Object> predicate;
049        /** The parameter to pass to {@code predicate}. */
050        private final Object param;
051    
052        /**
053         * Create a new BoundPredicate instance.
054         * @param <A> input type
055         * @param predicate the predicate to adapt
056         * @param arg the constant argument to use
057         */
058        @SuppressWarnings("unchecked")
059        public <A> BoundPredicate(UnaryPredicate<? super A> predicate, A arg) {
060            this.predicate =
061                (UnaryPredicate<Object>) Validate.notNull(predicate,
062                    "UnaryPredicate argument was null");
063            this.param = arg;
064        }
065    
066        /**
067         * {@inheritDoc}
068         */
069        public boolean test() {
070            return predicate.test(param);
071        }
072    
073        /**
074         * {@inheritDoc}
075         */
076        @Override
077        public boolean equals(Object that) {
078            return that == this || (that instanceof BoundPredicate && equals((BoundPredicate) that));
079        }
080    
081        /**
082         * Learn whether another BoundPredicate is equal to this.
083         * @param that BoundPredicate to test
084         * @return boolean
085         */
086        public boolean equals(BoundPredicate that) {
087            return null != that
088                    && (null == predicate ? null == that.predicate : predicate.equals(that.predicate))
089                    && (null == param ? null == that.param : param.equals(that.param));
090    
091        }
092    
093        /**
094         * {@inheritDoc}
095         */
096        @Override
097        public int hashCode() {
098            int hash = "BoundPredicate".hashCode();
099            if (null != predicate) {
100                hash <<= 2;
101                hash ^= predicate.hashCode();
102            }
103            if (null != param) {
104                hash <<= 2;
105                hash ^= param.hashCode();
106            }
107            return hash;
108        }
109    
110        /**
111         * {@inheritDoc}
112         */
113        @Override
114        public String toString() {
115            return "BoundPredicate<" + predicate + "(" + param + ")>";
116        }
117    
118        /**
119         * Adapt the given, possibly-<code>null</code>,
120         * {@link UnaryPredicate UnaryPredicate} to the
121         * {@link Predicate Predicate} interface by binding
122         * the specified <code>Object</code> as a constant
123         * argument.
124         * When the given <code>UnaryPredicate</code> is <code>null</code>,
125         * returns <code>null</code>.
126         *
127         * @param <A> input type
128         * @param predicate the possibly-<code>null</code>
129         *        {@link UnaryPredicate UnaryPredicate} to adapt
130         * @param arg the object to bind as a constant argument
131         * @return a <code>BoundPredicate</code> wrapping the given
132         *         {@link UnaryPredicate UnaryPredicate}, or <code>null</code>
133         *         if the given <code>UnaryPredicate</code> is <code>null</code>
134         */
135        public static <A> BoundPredicate bind(UnaryPredicate<? super A> predicate, A arg) {
136            return null == predicate ? null : new BoundPredicate(predicate, arg);
137        }
138    
139    }
