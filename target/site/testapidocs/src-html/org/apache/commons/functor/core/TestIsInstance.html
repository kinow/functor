017    package org.apache.commons.functor.core;
018    
019    import static org.junit.Assert.assertEquals;
020    import static org.junit.Assert.assertFalse;
021    import static org.junit.Assert.assertTrue;
022    import static org.junit.Assert.fail;
023    
024    import org.apache.commons.functor.BaseFunctorTest;
025    import org.apache.commons.functor.BinaryPredicate;
026    import org.apache.commons.functor.UnaryPredicate;
027    import org.junit.Test;
028    
029    /**
030     * @version $Revision: 1187622 $ $Date: 2011-10-21 23:22:33 -0200 (Fri, 21 Oct 2011) $
031     * @author Rodney Waldhoff
032     */
033    public class TestIsInstance extends BaseFunctorTest {
034    
035        // Functor Testing Framework
036        // ------------------------------------------------------------------------
037    
038        @Override
039        protected Object makeFunctor() {
040            return IsInstance.of(String.class);
041        }
042    
043        // Tests
044        // ------------------------------------------------------------------------
045    
046        @Test
047        public void testTest() throws Exception {
048            BinaryPredicate<Object, Class<?>> p = IsInstance.INSTANCE;
049            assertFalse(p.test(null, Number.class));
050            assertFalse(p.test("foo", Number.class));
051            assertTrue(p.test(3, Number.class));
052            assertTrue(p.test(3L, Number.class));
053        }
054    
055        @Test
056        public void testBoundTest() throws Exception {
057            UnaryPredicate<Object> p = IsInstance.of(Number.class);
058            assertFalse(p.test(null));
059            assertFalse(p.test("foo"));
060            assertTrue(p.test(3));
061            assertTrue(p.test(3L));
062        }
063    
064        @Test
065        public void testInstanceOfNull() throws Exception {
066            BinaryPredicate<Object, Class<?>> p = new IsInstance<Object>();
067            try {
068                p.test("foo", null);
069                fail("Expected NullPointerException");
070            } catch(NullPointerException e) {
071                // expected
072            }
073        }
074    
075        @Test
076        public void testEquals() throws Exception {
077            BinaryPredicate<Object, Class<?>> p = IsInstance.INSTANCE;
078            assertEquals(p, p);
079            assertObjectsAreEqual(p, IsInstance.instance());
080            assertObjectsAreNotEqual(p,Constant.truePredicate());
081        }
082    
083        @Test
084        public void testBoundEquals() throws Exception {
085            UnaryPredicate<Object> p = IsInstance.of(Object.class);
086            assertEquals(p,p);
087            assertObjectsAreEqual(p,IsInstance.of(Object.class));
088            assertObjectsAreNotEqual(p,Constant.truePredicate());
089            assertObjectsAreNotEqual(p,IsInstance.of(null));
090            assertObjectsAreNotEqual(p,IsInstance.of(String.class));
091        }
092    }
