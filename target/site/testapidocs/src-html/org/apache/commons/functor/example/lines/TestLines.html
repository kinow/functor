019    import java.io.Reader;

025    import junit.framework.TestSuite;
026    
027    import org.apache.commons.functor.adapter.ProcedureUnaryProcedure;
028    import org.apache.commons.functor.core.Offset;
029    import org.apache.commons.functor.core.algorithm.FoldLeft;
030    import org.apache.commons.functor.core.collection.Size;
031    import org.apache.commons.functor.core.composite.UnaryAnd;
032    import org.apache.commons.functor.core.composite.UnaryNot;
033    import org.apache.commons.functor.generator.FilteredGenerator;
034    import org.apache.commons.functor.generator.TransformedGenerator;

036    /**
037     * @version $Revision: 1187622 $ $Date: 2011-10-21 23:22:33 -0200 (Fri, 21 Oct 2011) $
038     * @author Rodney Waldhoff
039     */
040    public class TestLines extends TestCase {
041    
042        public TestLines(String testName) {
043            super(testName);
044        }
045    
046        public static Test suite() {
047            return new TestSuite(TestLines.class);
048        }
049    
050        private Reader reader = null;
051    
052        @Override
053        public void setUp() throws Exception {
054            super.setUp();
055            reader = new StringReader(DOCUMENT);
056        }
057    
058        @Override
059        public void tearDown() throws Exception {
060            super.tearDown();
061            reader = null;
062        }
063    
064        public void testCountCharacters() throws Exception {
065            Object result = new FoldLeft<Integer>(Sum.instance()).evaluate(
066                    new TransformedGenerator<String, Integer>(Lines.from(reader), new Size<String>()));
067    
068            assertEquals("Expected 990 characters",new Integer(990),result);
069        }
070    
071        public void testCountWords() throws Exception {
072            Object result = new FoldLeft<Integer>(Sum.instance()).evaluate(
073                    new TransformedGenerator<String, Integer>(Lines.from(reader),WordCount.instance()));
074    
075            assertEquals("Expected 157 words",new Integer(157),result);
076        }
077    
078        public void testCountLines() throws Exception {
079            Count count = new Count();
080            Lines
081                .from(reader)
082                    .run(ProcedureUnaryProcedure.adapt(count));
083    
084            assertEquals("Expected 16 lines",16,count.getCount());
085        }
086    
087        public void testCountWordsExcludingComments() throws Exception {
088            Object result = new FoldLeft<Integer>(Sum.instance()).evaluate(new TransformedGenerator<String, Integer>(
089                    new FilteredGenerator<String>(Lines.from(reader), UnaryNot.not(new StartsWith<String>("#"))), WordCount
090                            .instance()));
091    
092            assertEquals("Expected 90 words",new Integer(90),result);
093        }
094    
095        public void testCountCommentLines() throws Exception {
096            Count count = new Count();
097            new FilteredGenerator<String>(Lines.from(reader), new StartsWith<String>("#"))
098                        .run(ProcedureUnaryProcedure.<String>adapt(count));
099    
100            assertEquals("Expected 6 lines",6,count.getCount());
101        }
102    
103        public void testFindMatchingLines() throws Exception {
104            Collection<String> matches = new FilteredGenerator<String>(Lines.from(reader), new Contains<String>("lo"))
105                    .toCollection();
106            assertEquals("Expected 5 lines",5,matches.size());
107        }
108    
109        public void testFindMatchingFromTail() throws Exception {
110            Collection<String> matches = new FilteredGenerator<String>(Lines.from(reader), new UnaryAnd<String>(new Offset(
111                    8), new Contains<String>("lo"))).toCollection();
112            assertEquals("Expected 2 lines",2,matches.size());
113        }
