017    package org.apache.commons.functor.example.kata.four;
018    
019    import java.io.InputStream;
020    import java.io.InputStreamReader;
021    import java.io.Reader;
022    
023    import org.apache.commons.functor.BinaryFunction;
024    import org.apache.commons.functor.UnaryFunction;
025    import org.apache.commons.functor.adapter.BinaryFunctionUnaryFunction;
026    import org.apache.commons.functor.core.IsNull;
027    import org.apache.commons.functor.core.LeftIdentity;
028    import org.apache.commons.functor.core.RightIdentity;
029    import org.apache.commons.functor.core.algorithm.FoldLeft;
030    import org.apache.commons.functor.core.comparator.IsLessThan;
031    import org.apache.commons.functor.core.composite.Composite;
032    import org.apache.commons.functor.core.composite.Conditional;
033    import org.apache.commons.functor.core.composite.ConditionalBinaryFunction;
034    import org.apache.commons.functor.example.kata.one.Subtract;
035    import org.apache.commons.functor.example.lines.Lines;
036    import org.apache.commons.functor.generator.FilteredGenerator;
037    
038    /**
039     * The real workhorse of this Kata excercise.
040     *
041     * DataMunger wires together various functors and exposes them
042     * as static utility methhods.
043     * @version $Revision: 1171267 $ $Date: 2011-09-15 17:46:08 -0300 (Thu, 15 Sep 2011) $
044     * @author Rodney Waldhoff
045     */
046    public class DataMunger {
047        /** See {@link #process(Reader,int,int,int)} */
048        public static final Object process(final InputStream file, final int selected, final int col1, final int col2) {
049            return process(new InputStreamReader(file),selected,col1,col2);
050        }
051    
052        /**
053         * Processes each line of the given Reader, returning the <i>selected</i> column for the
054         * line where the absolute difference between the integer value of <i>col1</i> and <i>col2</i>
055         * is least.  Note that lines that don't begin with an Integer are ignored.
056         */
057        public static final Object process(final Reader file, final int selected, final int col1, final int col2) {
058            return NthColumn.instance(selected).evaluate(
059                    new FoldLeft<String>(lesserSpread(col1, col2)).evaluate(new FilteredGenerator<String>(Lines.from(file),
060                        Composite.predicate(IsInteger.instance(),NthColumn.instance(0)))));
061        }
062    
063        /**
064         * A BinaryFunction that will calculate the absolute
065         * difference between col1 and col2 in the given
066         * String arguments, and return the argument
067         * whose difference is smallest.
068         */
069        private static final BinaryFunction<String, String, String> lesserSpread(final int col1, final int col2) {
070            return new ConditionalBinaryFunction<String, String, String>(
071                IsNull.<String>left(),                                 // if left is null
072                RightIdentity.<String, String>function(),                      //   return right
073                Conditional.function(                          //   else return the parameter with the least spread
074                    Composite.predicate(                       //     if left is less than right
075                        IsLessThan.instance(),
076                        absSpread(col1,col2),
077                        absSpread(col1,col2)),
078                    LeftIdentity.<String, String>function(),                   //       return left
079                    RightIdentity.<String, String>function()                   //       else return right
080                )
081            );
082        }
083    
084        /**
085         * A UnaryFunction that returns the absolute value of the difference
086         * between the Integers stored in the <i>col1</i> and <i>col2</i>th
087         * whitespace delimited columns of the input line (a String).
088         */
089        private static UnaryFunction<String, Integer> absSpread(final int col1, final int col2) {
090            return Composite.function(
091                Abs.instance(),
092                new BinaryFunctionUnaryFunction<String, Number>(
093                    Composite.function(
094                        Subtract.instance(),
095                        Composite.function(ToInteger.instance(),NthColumn.instance(col1)),
096                        Composite.function(ToInteger.instance(),NthColumn.instance(col2)))
097                    ));
098        }
099    
100    }
