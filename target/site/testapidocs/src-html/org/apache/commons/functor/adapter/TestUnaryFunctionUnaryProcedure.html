017    package org.apache.commons.functor.adapter;
018    
019    import static org.junit.Assert.assertEquals;
020    import static org.junit.Assert.assertNotNull;
021    import static org.junit.Assert.assertNull;
022    
023    import org.apache.commons.functor.BaseFunctorTest;
024    import org.apache.commons.functor.UnaryFunction;
025    import org.apache.commons.functor.UnaryProcedure;
026    import org.apache.commons.functor.core.Constant;
027    import org.apache.commons.functor.core.NoOp;
028    import org.junit.Test;
029    
030    /**
031     * @version $Revision: 1188407 $ $Date: 2011-10-24 20:02:34 -0200 (Mon, 24 Oct 2011) $
032     * @author Rodney Waldhoff
033     */
034    public class TestUnaryFunctionUnaryProcedure extends BaseFunctorTest {
035    
036        // Functor Testing Framework
037        // ------------------------------------------------------------------------
038    
039        @Override
040        protected Object makeFunctor() {
041            return new UnaryFunctionUnaryProcedure<Object>(Constant.of("K"));
042        }
043    
044        // Tests
045        // ------------------------------------------------------------------------
046    
047        @Test
048        public void testRun() throws Exception {
049            class EvaluateCounter implements UnaryFunction<Object, Integer> {
050                int count = 0;
051                public Integer evaluate(Object a) { return Integer.valueOf(count++); }
052            }
053            EvaluateCounter counter = new EvaluateCounter();
054            UnaryProcedure<Object> p = new UnaryFunctionUnaryProcedure<Object>(counter);
055            assertEquals(0,counter.count);
056            p.run(null);
057            assertEquals(1,counter.count);
058            p.run("x");
059            assertEquals(2,counter.count);
060        }
061    
062        @Test
063        public void testEquals() throws Exception {
064            UnaryProcedure<Object> p = new UnaryFunctionUnaryProcedure<Object>(Constant.of("K"));
065            assertEquals(p,p);
066            assertObjectsAreEqual(p,new UnaryFunctionUnaryProcedure<Object>(Constant.of("K")));
067            assertObjectsAreNotEqual(p,NoOp.INSTANCE);
068            assertObjectsAreNotEqual(p,new UnaryFunctionUnaryProcedure<Object>(Constant.of("J")));
069        }
070    
071        @Test
072        public void testAdaptNull() throws Exception {
073            assertNull(UnaryFunctionUnaryProcedure.adapt(null));
074        }
075    
076        @Test
077        public void testAdapt() throws Exception {
078            assertNotNull(UnaryFunctionUnaryProcedure.adapt(Constant.of("K")));
079        }
080    }
