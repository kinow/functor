017    package org.apache.commons.functor.example.map;
018    
019    import java.util.HashMap;
020    import java.util.Map;
021    
022    import junit.framework.Test;
023    import junit.framework.TestCase;
024    import junit.framework.TestSuite;
025    
026    import org.apache.commons.functor.core.IsInstance;
027    
028    
029    /**
030     * @version $Revision: 1187622 $ $Date: 2011-10-21 23:22:33 -0200 (Fri, 21 Oct 2011) $
031     * @author Rodney Waldhoff
032     */
033    @SuppressWarnings("unchecked")
034    public class TestPredicatedMap extends TestCase {
035    
036        public TestPredicatedMap(String testName) {
037            super(testName);
038        }
039    
040        public static Test suite() {
041            return new TestSuite(TestPredicatedMap.class);
042        }
043    
044        private Map baseMap = null;
045        private Map predicatedMap = null;
046        @Override
047        public void setUp() throws Exception {
048            super.setUp();
049            baseMap = new HashMap();
050            predicatedMap = new PredicatedMap(baseMap,IsInstance.of(String.class),IsInstance.of(Integer.class));
051        }
052    
053        @Override
054        public void tearDown() throws Exception {
055            super.tearDown();
056            baseMap = null;
057            predicatedMap = null;
058        }
059    
060        // tests
061    
062        public void testCanPutMatchingPair() {
063            predicatedMap.put("xyzzy", new Integer(17));
064        }
065        public void testCantPutInvalidValue() {
066            try {
067                predicatedMap.put("xyzzy", "xyzzy");
068                fail("Expected IllegalArgumentException");
069            } catch(IllegalArgumentException e) {
070                // expected
071            }
072        }
073    
074        public void testCantPutInvalidKey() {
075            try {
076                predicatedMap.put(new Long(1), new Integer(3));
077                fail("Expected IllegalArgumentException");
078            } catch(IllegalArgumentException e) {
079                // expected
080            }
081        }
082    
083        public void testOnlyValidPairsAreAddedInPutAll() {
084            HashMap map = new HashMap();
085            map.put("one", new Integer(17));
086            map.put("two", "rejected");
087            map.put(new Integer(17), "xyzzy");
088            map.put(new Integer(7), new Integer(3));
089    
090            predicatedMap.putAll(map);
091            assertEquals(new Integer(17), predicatedMap.get("one"));
092            assertFalse(predicatedMap.containsKey("two"));
093    /*
094            assertFalse(predicatedMap.containsKey(new Integer(17)));
095            assertFalse(predicatedMap.containsKey(new Integer(7)));
096    */
097        }
098    }
