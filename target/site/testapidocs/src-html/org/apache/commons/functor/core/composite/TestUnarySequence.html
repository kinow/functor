017    package org.apache.commons.functor.core.composite;
018    
019    import static org.junit.Assert.assertEquals;
020    
021    import java.util.ArrayList;
022    import java.util.List;
023    
024    import org.apache.commons.functor.BaseFunctorTest;
025    import org.apache.commons.functor.UnaryProcedure;
026    import org.apache.commons.functor.core.NoOp;
027    import org.junit.Test;
028    
029    /**
030     * @version $Revision: 1187626 $ $Date: 2011-10-21 23:28:09 -0200 (Fri, 21 Oct 2011) $
031     * @author Rodney Waldhoff
032     */
033    @SuppressWarnings("unchecked")
034    public class TestUnarySequence extends BaseFunctorTest {
035    
036        // Functor Testing Framework
037        // ------------------------------------------------------------------------
038    
039        @Override
040        protected Object makeFunctor() {
041            return new UnarySequence<Object>(new NoOp(),new NoOp());
042        }
043    
044        // Tests
045        // ------------------------------------------------------------------------
046    
047        @Test
048        public void testRunZero() throws Exception {
049            UnarySequence<String> seq = new UnarySequence<String>();
050            seq.run(null);
051            seq.run("xyzzy");
052        }
053    
054        @Test
055        public void testRunOne() throws Exception {
056            RunCounter counter = new RunCounter();
057            UnarySequence<String> seq = new UnarySequence<String>(counter);
058            assertEquals(0,counter.count);
059            seq.run(null);
060            assertEquals(1,counter.count);
061            seq.run("xyzzy");
062            assertEquals(2,counter.count);
063        }
064    
065        @Test
066        public void testRunTwo() throws Exception {
067            RunCounter[] counter = { new RunCounter(), new RunCounter() };
068            UnarySequence<String> seq = new UnarySequence<String>(counter[0],counter[1]);
069            assertEquals(0,counter[0].count);
070            assertEquals(0,counter[1].count);
071            seq.run(null);
072            assertEquals(1,counter[0].count);
073            assertEquals(1,counter[1].count);
074            seq.run("xyzzy");
075            assertEquals(2,counter[0].count);
076            assertEquals(2,counter[1].count);
077        }
078    
079        @Test
080        public void testThen() throws Exception {
081            List<RunCounter> list = new ArrayList<RunCounter>();
082            UnarySequence<String> seq = new UnarySequence<String>();
083            seq.run(null);
084            for (int i=0;i<10;i++) {
085                RunCounter counter = new RunCounter();
086                seq.then(counter);
087                list.add(counter);
088                seq.run("xyzzy");
089                for (int j=0;j<list.size();j++) {
090                    assertEquals(list.size()-j,((list.get(j)).count));
091                }
092            }
093        }
094    
095        @Test
096        public void testEquals() throws Exception {
097            UnarySequence<?> p = new UnarySequence<Object>();
098            assertEquals(p,p);
099            UnarySequence<?> q = new UnarySequence<Object>();
100            assertObjectsAreEqual(p,q);
101    
102            for (int i=0;i<3;i++) {
103                p.then(new NoOp());
104                assertObjectsAreNotEqual(p,q);
105                q.then(new NoOp());
106                assertObjectsAreEqual(p,q);
107                p.then(new UnarySequence<Object>(new NoOp(),new NoOp()));
108                assertObjectsAreNotEqual(p,q);
109                q.then(new UnarySequence<Object>(new NoOp(),new NoOp()));
110                assertObjectsAreEqual(p,q);
111            }
112    
113            assertObjectsAreNotEqual(p,new NoOp());
114        }
115    
116        // Classes
117        // ------------------------------------------------------------------------
118    
119        static class RunCounter implements UnaryProcedure<Object> {
120            public void run(Object that) {
121                count++;
122            }
123            public int count = 0;
124        }
125    }
