017    package org.apache.commons.functor.example.kata.one;
018    
019    import org.apache.commons.functor.UnaryFunction;
020    
021    
022    /**
023     * @version $Revision: 665786 $ $Date: 2008-06-09 14:17:39 -0300 (Mon, 09 Jun 2008) $
024     * @author Rodney Waldhoff
025     */
026    public class Product {
027        public Product(String name, String sku, int cost) {
028            this(name,sku,ToMoney.from(Multiply.by(cost)));
029        }
030    
031        public Product(String name, String sku, UnaryFunction<? super Integer, Money> price) {
032            this.name = name;
033            this.sku = sku;
034            this.priceFunction = price;
035        }
036    
037        public String getName() {
038            return name;
039        }
040    
041        public UnaryFunction<? super Integer, Money> getPriceFunction() {
042            return priceFunction;
043        }
044    
045        public String getSku() {
046            return sku;
047        }
048    
049        public void setName(String string) {
050            name = string;
051        }
052    
053        public void setPriceFunction(UnaryFunction<? super Integer, Money> function) {
054            priceFunction = function;
055        }
056    
057        public void setSku(String string) {
058            sku = string;
059        }
060    
061        public Money getPrice(int quantity) {
062            return priceFunction.evaluate(quantity);
063        }
064    
065        private String name;
066        private String sku;
067        private UnaryFunction<? super Integer, Money> priceFunction;
068    }
