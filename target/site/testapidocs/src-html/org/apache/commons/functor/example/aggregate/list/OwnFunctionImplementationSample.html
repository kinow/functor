017    package org.apache.commons.functor.example.aggregate.list;
018    
019    import java.util.List;
020    
021    import org.apache.commons.functor.UnaryFunction;
022    import org.apache.commons.functor.aggregate.ArrayListBackedAggregator;
023    import org.junit.Test;
024    import static org.junit.Assert.assertEquals;
025    
026    /**
027     * Shows how to implement own aggregation function with a List-backed
028     * aggregator. In this example we want to monitor whether a certain value
029     * appears in the list or not.
030     */
031    public class OwnFunctionImplementationSample {
032        @Test
033        public void findValue() throws Exception {
034            // we're looking for 10
035            ArrayListBackedAggregator<Integer> agg = new ArrayListBackedAggregator<Integer>(new OwnFunction(10));
036            int eval = agg.evaluate();
037            assertEquals(eval, -1);
038    
039            agg.add( 1 );
040            eval = agg.evaluate();
041            assertEquals(eval, -1);
042    
043            agg.add( 2 );
044            eval = agg.evaluate();
045            assertEquals(eval, -1);
046    
047            agg.add( 10 );
048            eval = agg.evaluate();
049            assertEquals(eval, 2);
050            //function finds FIRST occurence!
051            agg.add( 10 );
052            eval = agg.evaluate();
053            assertEquals(eval, 2);
054            agg.add( 10 );
055            eval = agg.evaluate();
056            assertEquals(eval, 2);
057            agg.add( 10 );
058            eval = agg.evaluate();
059            assertEquals(eval, 2);
060        }
061    
062        /**
063         * This function returns the index of the first occurrence in the list of
064         * the given value.
065         */
066        static class OwnFunction implements UnaryFunction<List<Integer>, Integer> {
067            /** Value to find in the list. */
068            private int valueToFind;
069    
070            public OwnFunction(int valueToFind) {
071                this.valueToFind = valueToFind;
072            }
073    
074            /**
075             * Search in the list and find the first index of the given value.
076             *
077             * @param lst
078             *            List to search in
079             * @return index of {@link #valueToFind} if found or -1 if value not
080             *         present in the list
081             */
082            public Integer evaluate(List<Integer> lst) {
083                if (lst == null || lst.size() == 0) {
084                    return -1;
085                }
086                for (int i = 0; i < lst.size(); i++) {
087                    if (lst.get(i).intValue() == valueToFind) {
088                        return i;
089                    }
090                }
091                return -1;
092            }
093        }
094    }
