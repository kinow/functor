017    package org.apache.commons.functor.generator.util;
018    
019    import static org.junit.Assert.assertEquals;
020    import static org.junit.Assert.assertNull;
021    
022    import java.util.ArrayList;
023    import java.util.Collection;
024    import java.util.HashMap;
025    import java.util.Iterator;
026    import java.util.List;
027    import java.util.Map;
028    
029    import org.apache.commons.functor.BaseFunctorTest;
030    import org.apache.commons.functor.core.Limit;
031    import org.apache.commons.functor.core.Offset;
032    import org.apache.commons.functor.generator.GenerateUntil;
033    import org.apache.commons.functor.generator.GenerateWhile;
034    import org.apache.commons.functor.generator.UntilGenerate;
035    import org.apache.commons.functor.generator.WhileGenerate;
036    import org.junit.Before;
037    import org.junit.Test;
038    
039    /**
040     * @author Jason Horman (jason@jhorman.org)
041     */
042    public class TestEachElement extends BaseFunctorTest {
043    
044        private List<Integer> list = null;
045        private Map<String, String> map = null;
046        private Object[] array = null;
047    
048        @Override
049        protected Object makeFunctor() throws Exception {
050            return EachElement.from(new ArrayList<Object>());
051        }
052    
053        // Lifecycle
054        // ------------------------------------------------------------------------
055    
056        @Before
057        public void setUp() throws Exception {
058            list = new ArrayList<Integer>();
059            list.add(new Integer(0));
060            list.add(new Integer(1));
061            list.add(new Integer(2));
062            list.add(new Integer(3));
063            list.add(new Integer(4));
064    
065            map = new HashMap<String, String>();
066            map.put("1", "1-1");
067            map.put("2", "2-1");
068            map.put("3", "3-1");
069            map.put("4", "4-1");
070            map.put("5", "5-1");
071    
072            array = new String[5];
073            array[0] = "1";
074            array[1] = "2";
075            array[2] = "3";
076            array[3] = "4";
077            array[4] = "5";
078        }
079    
080        // Tests
081        // ------------------------------------------------------------------------
082    
083        @Test
084        public void testFromNull() {
085            assertNull(EachElement.from((Collection<?>) null));
086            assertNull(EachElement.from((Map<?, ?>) null));
087            assertNull(EachElement.from((Iterator<?>) null));
088            assertNull(EachElement.from((Object[]) null));
089        }
090    
091    
092        @Test
093        public void testWithList() {
094            Collection<?> col = EachElement.from(list).toCollection();
095            assertEquals("[0, 1, 2, 3, 4]", col.toString());
096        }
097    
098        @Test
099        public void testWithMap() {
100            List<?> col = (List<?>) EachElement.from(map).toCollection();
101            int i = 0;
102            for (;i<col.size();i++) {
103                Map.Entry entry = (Map.Entry) col.get(i);
104                if (entry.getKey().equals("1")) {
105                    assertEquals("1-1", entry.getValue());
106                } else if (entry.getKey().equals("2")) {
107                    assertEquals("2-1", entry.getValue());
108                } else if (entry.getKey().equals("3")) {
109                    assertEquals("3-1", entry.getValue());
110                } else if (entry.getKey().equals("4")) {
111                    assertEquals("4-1", entry.getValue());
112                } else if (entry.getKey().equals("5")) {
113                    assertEquals("5-1", entry.getValue());
114                }
115            }
116    
117            assertEquals(5, i);
118        }
119    
120        @Test
121        public void testWithArray() {
122            Collection<?> col = EachElement.from(array).toCollection();
123            assertEquals("[1, 2, 3, 4, 5]", col.toString());
124        }
125    
126        @Test
127        public void testWithStop() {
128            assertEquals("[0, 1, 2]", new UntilGenerate<Integer>(new Offset(3), EachElement.from(list)).toCollection().toString());
129            assertEquals("[0, 1, 2, 3]", new GenerateUntil<Integer>(EachElement.from(list), new Offset(3)).toCollection().toString());
130            assertEquals("[0, 1, 2]", new WhileGenerate<Integer>(new Limit(3), EachElement.from(list)).toCollection().toString());
131            assertEquals("[0, 1, 2, 3]", new GenerateWhile<Integer>(EachElement.from(list), new Limit(3)).toCollection().toString());
132        }
133    
134        @Test
135        public void testWithIterator() {
136            Collection<?> col = EachElement.from(list.iterator()).toCollection();
137            assertEquals("[0, 1, 2, 3, 4]", col.toString());
138        }
139    
140    }
