017    package org.apache.commons.functor.core.algorithm;
018    
019    import static org.junit.Assert.assertEquals;
020    import static org.junit.Assert.assertTrue;
021    
022    import org.apache.commons.functor.Procedure;
023    import org.apache.commons.functor.core.Offset;
024    import org.apache.commons.functor.core.algorithm.UntilDo;
025    import org.junit.Test;
026    
027    /**
028     * Tests {@link UntilDo} algorithm.
029     */
030    public class TestUntilDo {
031    
032        @Test
033        public final void testObjectEquals() throws Exception {
034            Counter counter = new Counter();
035            Object obj = new UntilDo(new Offset(10), counter);
036            assertEquals("equals must be reflexive",obj,obj);
037            assertEquals("hashCode must be reflexive",obj.hashCode(),obj.hashCode());
038            assertTrue(! obj.equals(null) ); // should be able to compare to null
039    
040            Object obj2 = new UntilDo(new Offset(10), counter);
041            if (obj.equals(obj2)) {
042                assertEquals("equals implies hash equals",obj.hashCode(),obj2.hashCode());
043                assertEquals("equals must be symmetric",obj2,obj);
044            } else {
045                assertTrue("equals must be symmetric",! obj2.equals(obj));
046            }
047        }
048    
049        @Test
050        public void testUntilDo() {
051            for (int i=0;i<3;i++){
052                Counter counter = new Counter();
053                new UntilDo(new Offset(i), counter).run();
054                assertEquals(i,counter.count);
055            }
056        }
057    
058        // Classes
059        // ------------------------------------------------------------------------
060    
061        static class Counter implements Procedure {
062            public void run() {
063                count++;
064            }
065            public int count = 0;
066        }
067    
068    }
