017    package org.apache.commons.functor.example.kata.two;
018    
019    import org.apache.commons.functor.Function;
020    import org.apache.commons.functor.Predicate;
021    import org.apache.commons.functor.Procedure;
022    import org.apache.commons.functor.core.Constant;
023    import org.apache.commons.functor.core.NoOp;
024    
025    /**
026     * Supports an Eiffel style loop construct.
027     * <pre>
028     * new EiffelStyleLoop()
029     *   .from(new Procedure() { public void run() {} }) // init code
030     *   .invariant(new Predicate() { public boolean test() {} }) // invariants
031     *   .variant(new Procedure() { public Object evaluate() {} }) // diminishing comparable value
032     *   // or
033     *   // .variant(new Predicate() { public boolean test() {} }) // more invariants
034     *   .until(new Predicate() { public boolean test() {} }) // terminating condition
035     *   .loop(new Procedure() { public void run() {} }) // the acutal loop
036     *   .run();
037     * </pre>
038     *
039     * Note that <tt>new EiffelStyleLoop().run()</tt> executes just fine.
040     * You only need to set the parts of the loop you want to use.
041     *
042     * @version $Revision: 666191 $ $Date: 2008-06-10 13:43:53 -0300 (Tue, 10 Jun 2008) $
043     * @author Rodney Waldhoff
044     */
045    public class EiffelStyleLoop implements Procedure {
046        public EiffelStyleLoop from(Procedure procedure) {
047            from = procedure;
048            return this;
049        }
050    
051        public EiffelStyleLoop invariant(Predicate predicate) {
052            invariant = predicate;
053            return this;
054        }
055    
056        public EiffelStyleLoop variant(Predicate predicate) {
057            variant = predicate;
058            return this;
059        }
060    
061        @SuppressWarnings("unchecked")
062        public EiffelStyleLoop variant(final Function function) {
063            return variant(new Predicate() {
064                public boolean test() {
065                    boolean result = true;
066                    Comparable next = (Comparable)(function.evaluate());
067                    if (null != last) {
068                        result = last.compareTo(next) > 0;
069                    }
070                    last = next;
071                    return result;
072                }
073                private Comparable last = null;
074            });
075        }
076    
077        public EiffelStyleLoop until(Predicate predicate) {
078            until = predicate;
079            return this;
080        }
081    
082        public EiffelStyleLoop loop(Procedure procedure) {
083            loop = procedure;
084            return this;
085        }
086    
087        public void run() {
088            from.run();
089            assertTrue(invariant.test());
090            while(! until.test() ) {
091                loop.run();
092                assertTrue(variant.test());
093                assertTrue(invariant.test());
094            }
095    
096            // Note that:
097            //   assertTrue(until.test());
098            // holds here, but isn't necessary since that's
099            // the only way we could get out of the loop
100    
101            // Also note that:
102            //   assertTrue(invariant.test());
103            // holds here, but was the last thing called
104            // before until.test()
105        }
106    
107        private void assertTrue(boolean value) {
108            if (!value) {
109                throw new IllegalStateException("Assertion failed");
110            }
111        }
112    
113        private Procedure from = NoOp.instance();
114        private Predicate invariant = Constant.truePredicate();
115        private Predicate variant = Constant.truePredicate();
116        private Predicate until = Constant.falsePredicate();
117        private Procedure loop = NoOp.instance();
118    
119    }
