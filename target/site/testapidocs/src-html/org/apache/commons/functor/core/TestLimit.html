017    package org.apache.commons.functor.core;
018    
019    import static org.junit.Assert.assertTrue;
020    import static org.junit.Assert.fail;
021    
022    import org.apache.commons.functor.BaseFunctorTest;
023    import org.apache.commons.functor.BinaryPredicate;
024    import org.apache.commons.functor.Predicate;
025    import org.apache.commons.functor.UnaryPredicate;
026    import org.junit.Test;
027    
028    /**
029     * @version $Revision: 1187622 $ $Date: 2011-10-21 23:22:33 -0200 (Fri, 21 Oct 2011) $
030     * @author Rodney Waldhoff
031     */
032    public class TestLimit extends BaseFunctorTest {
033    
034    
035        // Functor Testing Framework
036        // ------------------------------------------------------------------------
037    
038        @Override
039        protected Object makeFunctor() {
040            return new Limit(3);
041        }
042    
043        // Tests
044        // ------------------------------------------------------------------------
045    
046        @Test
047        public void testZero() throws Exception {
048            Predicate p = new Limit(0);
049            assertTrue(! p.test());
050            assertTrue(! p.test());
051            assertTrue(! p.test());
052        }
053    
054        @Test
055        public void testBadArgs() throws Exception {
056            try {
057                new Limit(-1);
058                fail("Expected IllegalArgumentException");
059            } catch(IllegalArgumentException e) {
060                // expected
061            }
062        }
063    
064        @Test
065        public void testTestNullary() throws Exception {
066            Predicate p = new Limit(3);
067            assertTrue(p.test());
068            assertTrue(p.test());
069            assertTrue(p.test());
070            assertTrue(! p.test());
071        }
072    
073        @Test
074        public void testTestUnary() throws Exception {
075            UnaryPredicate<Object> p = new Limit(3);
076            assertTrue(p.test(null));
077            assertTrue(p.test(null));
078            assertTrue(p.test(null));
079            assertTrue(! p.test(null));
080        }
081    
082        @Test
083        public void testTestBinary() throws Exception {
084            BinaryPredicate<Object, Object> p = new Limit(3);
085            assertTrue(p.test(null,null));
086            assertTrue(p.test(null,null));
087            assertTrue(p.test(null,null));
088            assertTrue(! p.test(null,null));
089        }
090    }
