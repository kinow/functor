017    package org.apache.commons.functor.example.kata.four;
018    
019    import java.util.StringTokenizer;
020    
021    import org.apache.commons.functor.UnaryFunction;
022    
023    /**
024     * Evaluates the input String to extrace the nth whitespace
025     * delmited column.
026     *
027     * @version $Revision: 1187624 $ $Date: 2011-10-21 23:25:48 -0200 (Fri, 21 Oct 2011) $
028     * @author Rodney Waldhoff
029     */
030    public final class NthColumn implements UnaryFunction<String, String> {
031        public NthColumn(int n) {
032            this.n = n;
033        }
034    
035        public String evaluate(String obj) {
036            StringTokenizer toker = new StringTokenizer(obj);
037            for (int count = 0; count < n && toker.hasMoreTokens();count++) {
038                toker.nextToken();
039            }
040            return toker.hasMoreTokens() ? toker.nextToken() : null;
041        }
042    
043        private final int n;
044    
045        public static final NthColumn instance(int n) {
046            return new NthColumn(n);
047        }
048    }
