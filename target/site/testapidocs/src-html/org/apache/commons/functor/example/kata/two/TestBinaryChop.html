032    /**
033     * Examples of binary search implementations.
034     *
035     * A binary search algorithm is the same strategy used in
036     * that number guessing game, where one player picks a number
037     * between 1 and 100, and the second player tries to guess it.
038     * Each time the second player guesses a number, the first player
039     * tells whether the chosen number is higher, lower or equal to
040     * the guess.
041     *
042     * An effective strategy for this sort of game is always guess
043     * the midpoint between what you know to be the lowest and
044     * highest possible number.  This will find the number in
045     * log2(N) guesses (when N = 100, this is at most 7 guesses).
046     *
047     * For example, suppose the first player (secretly) picks the
048     * number 63.  The guessing goes like this:
049     *
050     * P1> I'm thinking of a number between 1 and 100.
051     * P2> Is it 50?
052     * P1> Higher.
053     * P2> 75?
054     * P1> Lower.
055     * P2> 62?
056     * P1> Higher.
057     * P2> 68?
058     * P1> Lower.
059     * P2> 65?
060     * P1> Lower.
061     * P2> 63?
062     * P1> That's it.
063     *
064     * Dave Thomas's Kata Two asks us to implement a binary search algorithm
065     * in several ways.  Here we'll use this as an opportunity to
066     * consider some common approaches and explore
067     * some functor based approaches as well.
068     *
069     * See http://pragprog.com/pragdave/Practices/Kata/KataTwo.rdoc,v
070     * for more information on this Kata.
071     *
072     * @version $Revision: 1187622 $ $Date: 2011-10-21 23:22:33 -0200 (Fri, 21 Oct 2011) $
073     * @author Rodney Waldhoff
074     */

080         * a quick check of searching a fairly large
081         * list, just to make sure the time and space
082         * requirements are reasonable.
083         */
084        private void chopTest(BinaryChop chopper) {
085            assertEquals(-1, chopper.find(3, new int[0]));
086            assertEquals(-1, chopper.find(3, new int[] { 1 }));
087            assertEquals(0, chopper.find(1, new int[] { 1 }));
088    
089            assertEquals(0, chopper.find(1, new int[] { 1, 3, 5 }));
090            assertEquals(1, chopper.find(3, new int[] { 1, 3, 5 }));
091            assertEquals(2, chopper.find(5, new int[] { 1, 3, 5 }));
092            assertEquals(-1, chopper.find(0, new int[] { 1, 3, 5 }));
093            assertEquals(-1, chopper.find(2, new int[] { 1, 3, 5 }));
094            assertEquals(-1, chopper.find(4, new int[] { 1, 3, 5 }));
095            assertEquals(-1, chopper.find(6, new int[] { 1, 3, 5 }));
096    
097            assertEquals(0, chopper.find(1, new int[] { 1, 3, 5, 7 }));
098            assertEquals(1, chopper.find(3, new int[] { 1, 3, 5, 7 }));
099            assertEquals(2, chopper.find(5, new int[] { 1, 3, 5, 7 }));
100            assertEquals(3, chopper.find(7, new int[] { 1, 3, 5, 7 }));
101            assertEquals(-1, chopper.find(0, new int[] { 1, 3, 5, 7 }));
102            assertEquals(-1, chopper.find(2, new int[] { 1, 3, 5, 7 }));
103            assertEquals(-1, chopper.find(4, new int[] { 1, 3, 5, 7 }));
104            assertEquals(-1, chopper.find(6, new int[] { 1, 3, 5, 7 }));
105            assertEquals(-1, chopper.find(8, new int[] { 1, 3, 5, 7 }));
106    
107            List largeList = (List) (new IntegerRange(0, 100001).toCollection());
108            assertEquals(-1, chopper.find(new Integer(-5), largeList));
109            assertEquals(100000, chopper.find(new Integer(100000), largeList));
110            assertEquals(0, chopper.find(new Integer(0), largeList));
111            assertEquals(50000, chopper.find(new Integer(50000), largeList));

139        /**
140         * Here's a basic iterative approach.
141         *
142         * We set the lower or upper bound to the midpoint
143         * until there's only one element between the lower
144         * and upper bound.  Then the lower bound is where
145         * the element would be found if it existed in the
146         * list.
147         *
148         * We add an additional comparision at the end so
149         * that we can return -1 if the element is not yet
150         * in the list.
151         */
152        @Test
153        public void testIterative() {
154            chopTest(new BaseBinaryChop() {
155                @Override
156                public int find(Object seeking, List list) {
157                    int high = list.size();
158                    int low = 0;
159                    while (high - low > 1) {
160                        int mid = (high + low) / 2;
161                        if (greaterThan(list,mid,seeking)) {
162                            high = mid;
163                        } else {
164                            low = mid;
165                        }
166                    }
167                    return list.isEmpty() ? -1 : (equals(list,low,seeking) ? low : -1);
168                }
169            });
170        }

172        /*
173         * At http://onestepback.org/index.cgi/Tech/Programming/Kata/KataTwoVariation1.rdoc,
174         * Jim Weirich discusses Kata Two from the perspective of loop invariants.
175         *
176         * Loop invariants provide a way of deductive reasoning about loops.
177         *
178         * Let P, Q. and R be predicates and A and B be
179         * procedures.  Note that if:
180         *   assert(P.test());
181         *   A.run();
182         *   assert(Q.test());
183         * and
184         *   assert(Q.test());
185         *   B.run();
186         *   assert(R.test());
187         * are both valid, then:
188         *   assert(P.test());
189         *   A.run();
190         *   B.run();
191         *   assert(R.test());
192         * is valid as well.
193         *
194         * Similiarly, if INV and TERM are predicates and BODY is a procedure,
195         * then if:
196         *   assert(INV.test());
197         *   BODY.run();
198         *   assert(INV.test());
199         * is valid, then so is:
200         *   assert(INV.test());
201         *   while(! TERM.test() ) { BODY.run(); }
202         *   assert(INV.test());
203         *   assert(TERM.test());
204         *
205         * Here INV is an "loop invariant", a statement that is true for every
206         * single iteration through the loop.  TERM is a terminating condition,
207         * a statement that is true (by construction) when the loop exits.
208         *
209         * We can use loop invariants to reason about our iterative binary
210         * search loop.  In particular, note that:
211         *
212         * // assert that the list is empty, or
213         * // the result index is between
214         * // low (inclusive) and high (exclusive),
215         * // or high is 0 (the list is empty)
216         * Predicate INV = new Predicate() {
217         *   public boolean test() {
218         *    return high == 0 ||
219         *          (low <= result && result < high);
220         *   }
221         * };
222         *
223         * is a valid invariant in our binary search, and that:
224         *
225         * Predicate TERM = new Predicate() {
226         *   public boolean test() {
227         *    return (high - low) <= 1;
228         *   }
229         * };
230         *
231         * is a valid terminating condition.
232         *
233         * The BODY in our case simply moves the endpoints
234         * closer together, without violating
235         * our invariant:
236         *
237         * Procedure BODY = new Procedure() {
238         *   public void run() {
239         *     int mid = (high + low) / 2;
240         *     if (greaterThan(list,mid,seeking)) {
241         *       high = mid;
242         *     } else {
243         *       low = mid;
244         *     }
245         *   }
246         * };
247         *
248         * One could assert our invariant before and after
249         * the execution of BODY, and the terminating condition
250         * at the end of the loop, but we can tell by construction
251         * that these assertions will hold.
252         *
253         * Using the functor framework, we can make these notions
254         * explict.  Specifically, the construction above is:
255         *
256         *   Algorithms.untildo(BODY,TERM);
257         *
258         * Since we'll want to share state among the TERM and BODY,
259         * let's declare a single interface for the TERM Predicate and
260         * the BODY Procedure.  We'll be calculating a result within
261         * the loop, so let's add a Function implementation as well,
262         * as a way of retrieving that result:
263         */
264        interface Loop extends Predicate, Procedure, Function {
265            /** The terminating condition. */
266            boolean test();
267            /** The loop body. */
268            void run();
269            /** The result of executing the loop. */
270            Object evaluate();
271        };

274         * Now we can use the Algorithms.dountil method to
275         * execute that loop:

277        @Test
278        public void testIterativeWithInvariants() {
279            chopTest(new BaseBinaryChop() {
280    
281                @Override
282                public int find(final Object seeking, final List list) {
283                    Loop loop = new Loop() {
284                        int high = list.size();
285                        int low = 0;
286    
287                        /** Our terminating condition. */
288                        public boolean test() {
289                            return (high - low) <= 1;
290                        }
291    
292                        /** Our loop body. */
293                        public void run() {
294                            int mid = (high + low) / 2;
295                            if (greaterThan(list,mid,seeking)) {
296                                high = mid;
297                            } else {
298                                low = mid;
299                            }
300                        }
301    
302                        /**
303                         * A way of returning the result
304                         * at the end of the loop.
305                         */
306                        public Object evaluate() {
307                            return new Integer(
308                                list.isEmpty() ?
309                                -1 :
310                                (BaseBinaryChop.equals(list,low,seeking) ? low : -1));
311                        }
312    
313                    };
314                    new UntilDo(loop, loop).run();
315                    return ((Number) loop.evaluate()).intValue();
316                }
317            });
318        }

320        /*
321         * Jim Weirich notes how Eiffel is very explict about loop invariants:
322         *
323         *   from
324         *     low := list.lower
325         *     high := list.upper + 1
326         *   invariant
327         *     lower_limit: -- low <= result (this is just a comment)
328         *     upper_limit: -- high < result (this is just a comment)
329         *   variant
330         *     high - low
331         *   until
332         *     (high - low) <= 1
333         *   loop
334         *     mid := (high + low) // 2
335         *     if list.at(mid) > seeking then
336         *       high := mid
337         *     else
338         *       low := mid
339         *     end
340         *   end
341         *
342         * We can do that too, using EiffelStyleLoop.
343         */
344        class BinarySearchLoop extends EiffelStyleLoop {
345            BinarySearchLoop(Object aSeeking, List aList) {
346                seeking = aSeeking;
347                list = aList;
348    
349                from(new Procedure() {
350                    public void run() {
351                        low = 0;
352                        high = list.size();
353                    }
354                });
355    
356                invariant(new Predicate() {
357                    public boolean test() {
358                        return high == 0 || low < high;
359                    }
360                });
361    
362                variant(new Function() {
363                    public Object evaluate() {
364                        return new Integer(high - low);
365                    }
366                });
367    
368                until(new Predicate() {
369                    public boolean test() {
370                        return high - low <= 1;
371                    }
372                });
373    
374                loop(new Procedure() {
375                    public void run() {
376                        int mid = (high + low) / 2;
377                        if (BaseBinaryChop.greaterThan(list,mid,seeking)) {
378                            high = mid;
379                        } else {
380                            low = mid;
381                        }
382                    }
383                });
384            }
385    
386            int getResult() {
387                return list.isEmpty() ? -1 : BaseBinaryChop.equals(list,low,seeking) ? low : -1;
388            }
389    
390            private int high;
391            private int low;
392            private final Object seeking;
393            private final List list;
394        }
395    
396        @Test
397        public void testIterativeWithInvariantsAndAssertions() {
398            chopTest(new BaseBinaryChop() {
399                @Override
400                public int find(Object seeking, List list) {
401                    BinarySearchLoop loop = new BinarySearchLoop(seeking,list);
402                    loop.run();
403                    return loop.getResult();
404                }});

407        /**
408         * A recursive version of that implementation uses
409         * method parameters to track the upper and
410         * lower bounds.
411         */
412        @Test
413        public void testRecursive() {
414            chopTest(new BaseBinaryChop() {
415                @Override
416                public int find(Object seeking, List list) {
417                    return find(seeking, list, 0, list.size());
418                }
419    
420                private int find(Object seeking, List list, int low, int high) {
421                    if (high - low > 1) {
422                        int mid = (high + low) / 2;
423                        if (greaterThan(list,mid,seeking)) {
424                            return find(seeking,list,low,mid);
425                        } else {
426                            return find(seeking,list,mid,high);
427                        }
428                    } else {
429                        return list.isEmpty() ? -1 : (equals(list,low,seeking) ? low : -1);
430                    }
431                }
432            });
433        }

435        /**
436         * We can use the Algorithms.recuse method
437         * to implement that as tail recursion.
438         *
439         * Here the anonymous Function implemenation
440         * holds this itermediate state, rather than
441         * the VM's call stack.
442         *
443         * Arguably this is more like a continuation than
444         * tail recursion, since there is a bit of state
445         * to be tracked.
446         */
447        @Test
448        public void testTailRecursive() {
449            chopTest(new BaseBinaryChop() {
450                @Override
451                public int find(final Object seeking, final List list) {
452                    return ((Number) new RecursiveEvaluation(new Function() {
453                        public Object evaluate() {
454                            if (high - low > 1) {
455                                int mid = (high + low) / 2;
456                                if (greaterThan(list,mid,seeking)) {
457                                    high = mid;
458                                } else {
459                                    low = mid;
460                                }
461                                return this;
462                            } else {
463                                return list.isEmpty() ?
464                                    BaseBinaryChop.NEGATIVE_ONE :
465                                    (BaseBinaryChop.equals(list,low,seeking) ?
466                                        new Integer(low) :
467                                        BaseBinaryChop.NEGATIVE_ONE);
468                            }
469                        }
470                        int high = list.size();
471                        int low = 0;
472                    }).evaluate()).intValue();
473                }
474            });
475        }

477        /**
478         * One fun functional approach is to "slice" up the
479         * list as we search,  looking at smaller and
480         * smaller slices until we've found the element
481         * we're looking for.
482         *
483         * Note that while any given call to this recursive
484         * function may only be looking at a sublist, we
485         * need to return the index in the overall list.
486         * Hence we'll split out a method so that we can
487         * pass the offset in the original list as a
488         * parameter.
489         *
490         * With all of the subList creation, this approach
491         * is probably less efficient than either the iterative
492         * or the recursive implemenations above.
493         */
494        @Test
495        public void testRecursive2() {
496            chopTest(new BaseBinaryChop() {
497                @Override
498                public int find(Object seeking, List list) {
499                    return find(seeking,list,0);
500                }
501    
502                private int find(Object seeking, List list, int offset) {
503                    if (list.isEmpty()) {
504                        return -1;
505                    } if (list.size() == 1) {
506                        return (equals(list,0,seeking) ? offset : -1);
507                    } else {
508                        int mid = list.size() / 2;
509                        if (greaterThan(list,mid,seeking)) {
510                            return find(seeking,list.subList(0,mid),offset);
511                        } else {
512                            return find(seeking,list.subList(mid,list.size()),offset+mid);
513                        }
514                    }
515                }
516            });
517        }

519        /**
520         * We can do that using tail recursion as well.
521         *
522         * Again, the anonymous Function implemenation
523         * holds the "continuation" state.
524         */
525        @Test
526        public void testTailRecursive2() {
527            chopTest(new BaseBinaryChop() {
528                @Override
529                public int find(final Object seeking, final List list) {
530                    return ((Number) new RecursiveEvaluation(new Function() {
531                        public Object evaluate() {
532                            if (sublist.isEmpty()) {
533                                return BaseBinaryChop.NEGATIVE_ONE;
534                            } if (sublist.size() == 1) {
535                                return (BaseBinaryChop.equals(sublist,0,seeking) ?
536                                    new Integer(offset) :
537                                    BaseBinaryChop.NEGATIVE_ONE);
538                            } else {
539                                int mid = sublist.size() / 2;
540                                if (greaterThan(sublist,mid,seeking)) {
541                                    sublist = sublist.subList(0,mid);
542                                } else {
543                                    sublist = sublist.subList(mid,sublist.size());
544                                    offset += mid;
545                                }
546                                return this;
547                            }
548                        }
549                        int offset = 0;
550                        List sublist = list;
551                    }).evaluate()).intValue();
552                }
553            });
554        }
