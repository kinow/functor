017    package org.apache.commons.functor.example.kata.one;
018    
019    import org.apache.commons.functor.UnaryFunction;
020    import org.apache.commons.functor.adapter.LeftBoundFunction;
021    
022    /**
023     * @version $Revision: 665786 $ $Date: 2008-06-09 14:17:39 -0300 (Mon, 09 Jun 2008) $
024     * @author Rodney Waldhoff
025     */
026    public class Multiply extends ArithmeticOperation {
027    
028        public Number evaluate(Number left, Number right) {
029            return new Integer(left.intValue() * right.intValue());
030        }
031    
032        public static Multiply instance() {
033            return INSTANCE;
034        }
035    
036        public static UnaryFunction<Number, Number> by(int factor) {
037            return LeftBoundFunction.bind(INSTANCE, factor);
038        }
039    
040        private static Multiply INSTANCE = new Multiply();
041    }
